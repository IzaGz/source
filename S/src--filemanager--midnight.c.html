1602 /**
1603  * Return a random hint.  If force is TRUE, ignore the timeout.
1604  */
1605 
1606 char *
1607 get_random_hint (gboolean force)
     /*  [+1607 src/filemanager/midnight.c] */
1608 {
1609     static const guint64 update_period = 60 * G_USEC_PER_SEC;
1610     static guint64 tv = 0;
1611 
1612     char *data, *result = NULL, *eop;
1613     size_t len, start;
1614     GIConv conv;
1615 
1616     /* Do not change hints more often than one minute */
1617     if (!force && !mc_time_elapsed (&tv, update_period))
1618         return g_strdup ("");
1619 
1620     data = load_mc_home_file (mc_global.share_data_dir, MC_HINT, NULL, &len);
1621     if (data == NULL)
1622         return NULL;
1623 
1624     /* get a random entry */
1625     srand ((unsigned int) (tv / G_USEC_PER_SEC));
1626     start = ((size_t) rand ()) % (len - 1);
1627 
1628     /* Search the start of paragraph */
1629     for (; start != 0; start--)
1630         if (data[start] == '\n' && data[start + 1] == '\n')
1631         {
1632             start += 2;
1633             break;
1634         }
1635 
1636     /* Search the end of paragraph */
1637     for (eop = data + start; *eop != '\0'; eop++)
1638     {
1639         if (*eop == '\n' && *(eop + 1) == '\n')
1640         {
1641             *eop = '\0';
1642             break;
1643         }
1644         if (*eop == '\n')
1645             *eop = ' ';
1646     }
1647 
1648     /* hint files are stored in utf-8 */
1649     /* try convert hint file from utf-8 to terminal encoding */
1650     conv = str_crt_conv_from ("UTF-8");
1651     if (conv != INVALID_CONV)
1652     {
1653         GString *buffer;
1654 
1655         buffer = g_string_sized_new (len - start);
1656         if (str_convert (conv, &data[start], buffer) != ESTR_FAILURE)
1657             result = g_string_free (buffer, FALSE);
1658         else
1659             g_string_free (buffer, TRUE);
1660         str_close_conv (conv);
1661     }
1662     else
1663         result = g_strndup (data + start, len - start);
1664 
1665     g_free (data);
1666     return result;
1667 }
