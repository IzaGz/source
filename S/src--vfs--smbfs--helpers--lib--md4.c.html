 146 static void
 147 copy64 (uint32 * M, unsigned char *in)
     /*  [+147 src/vfs/smbfs/helpers/lib/md4.c] */
 148 {
 149     int i;
 150 
 151     for (i = 0; i < 16; i++)
 152         M[i] = (in[i * 4 + 3] << 24) | (in[i * 4 + 2] << 16) |
 153             (in[i * 4 + 1] << 8) | (in[i * 4 + 0] << 0);

 156 static void
 157 copy4 (unsigned char *out, uint32 x)
     /*  [+157 src/vfs/smbfs/helpers/lib/md4.c] */
 158 {
 159     out[0] = x & 0xFF;
 160     out[1] = (x >> 8) & 0xFF;
