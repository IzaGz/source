 113 static int
 114 statfs (char const *filename, struct fs_info *buf)
     /*  [+114 src/filemanager/filegui.c] */
 115 {
 116     dev_t device = dev_for_path (filename);
 117 
 118     if (device < 0)
 119     {
 120         errno = (device == B_ENTRY_NOT_FOUND ? ENOENT
 121                  : device == B_BAD_VALUE ? EINVAL
 122                  : device == B_NAME_TOO_LONG ? ENAMETOOLONG
 123                  : device == B_NO_MEMORY ? ENOMEM : device == B_FILE_ERROR ? EIO : 0);
 124         return -1;
 125     }
 126     /* If successful, buf->dev will be == device.  */
 127     return fs_stat_dev (device, buf);
 128 }

 264 /* Return true if statvfs works.  This is false for statvfs on systems

 266    preceding entries in /proc/mounts; that makes df hang if even one

 271 statvfs_works (void)
     /*  [+271 src/filemanager/filegui.c] */
 272 {
 273 #if ! (__linux__ && (__GLIBC__ || __UCLIBC__))
 274     return 1;
 275 #else
 276     static int statvfs_works_cache = -1;
 277     struct utsname name;
 278 
 279     if (statvfs_works_cache < 0)
 280         statvfs_works_cache = (uname (&name) == 0 && 0 <= str_verscmp (name.release, "2.6.36"));
 281     return statvfs_works_cache;

 288 filegui__check_attrs_on_fs (const char *fs_path)
     /*  [+288 src/filemanager/filegui.c] */
 289 {
 290     STRUCT_STATVFS stfs;
 291 
 292     if (!copymove_persistent_attr)
 293         return FALSE;
 294 
 295 #if USE_STATVFS && defined(STAT_STATVFS)
 296     if (statvfs_works () && statvfs (fs_path, &stfs) != 0)
 297         return TRUE;
 298 #else
 299     if (STATFS (fs_path, &stfs) != 0)
 300         return TRUE;
 301 #endif
 302 
 303 #if (USE_STATVFS && defined(HAVE_STRUCT_STATVFS_F_TYPE)) || \
 304         (!USE_STATVFS && defined(HAVE_STRUCT_STATFS_F_TYPE))
 305     switch ((filegui_nonattrs_fs_t) stfs.f_type)
 306     {
 307     case MSDOS_SUPER_MAGIC:
 308     case NTFS_SB_MAGIC:
 309     case PROC_SUPER_MAGIC:
 310     case SMB_SUPER_MAGIC:
 311     case NCP_SUPER_MAGIC:
 312     case USBDEVICE_SUPER_MAGIC:
 313         return FALSE;
 314     default:
 315         break;
 316     }
 317 #elif defined(HAVE_STRUCT_STATVFS_F_FSTYPENAME) || defined(HAVE_STRUCT_STATFS_F_FSTYPENAME)
 318     if (strcmp (stfs.STATXFS_FILE_SYSTEM_TYPE_MEMBER_NAME, "msdos") == 0
 319         || strcmp (stfs.STATXFS_FILE_SYSTEM_TYPE_MEMBER_NAME, "msdosfs") == 0
 320         || strcmp (stfs.STATXFS_FILE_SYSTEM_TYPE_MEMBER_NAME, "ntfs") == 0
 321         || strcmp (stfs.STATXFS_FILE_SYSTEM_TYPE_MEMBER_NAME, "procfs") == 0
 322         || strcmp (stfs.STATXFS_FILE_SYSTEM_TYPE_MEMBER_NAME, "smbfs") == 0
 323         || strstr (stfs.STATXFS_FILE_SYSTEM_TYPE_MEMBER_NAME, "fusefs") != NULL)
 324         return FALSE;
 325 #elif defined(HAVE_STRUCT_STATVFS_F_BASETYPE)
 326     if (strcmp (stfs.STATXFS_FILE_SYSTEM_TYPE_MEMBER_NAME, "pcfs") == 0
 327         || strcmp (stfs.STATXFS_FILE_SYSTEM_TYPE_MEMBER_NAME, "ntfs") == 0
 328         || strcmp (stfs.STATXFS_FILE_SYSTEM_TYPE_MEMBER_NAME, "proc") == 0
 329         || strcmp (stfs.STATXFS_FILE_SYSTEM_TYPE_MEMBER_NAME, "smbfs") == 0
 330         || strcmp (stfs.STATXFS_FILE_SYSTEM_TYPE_MEMBER_NAME, "fuse") == 0)
 331         return FALSE;
 332 #endif
 333 
 334     return TRUE;
 335 }

 339 static void
 340 file_frmt_time (char *buffer, double eta_secs)
     /*  [+340 src/filemanager/filegui.c] */
 341 {
 342     int eta_hours, eta_mins, eta_s;
 343     eta_hours = (int) (eta_secs / (60 * 60));
 344     eta_mins = (int) ((eta_secs - (eta_hours * 60 * 60)) / 60);
 345     eta_s = (int) (eta_secs - (eta_hours * 60 * 60 + eta_mins * 60));
 346     g_snprintf (buffer, BUF_TINY, _("%d:%02d.%02d"), eta_hours, eta_mins, eta_s);
 347 }

 608 static gboolean
 609 is_wildcarded (const char *p)
     /*  [+609 src/filemanager/filegui.c] */
 610 {
 611     gboolean escaped = FALSE;
 612     for (; *p; p++)
 613     {
 614         if (*p == '\\')
 615         {
 616             if (p[1] >= '1' && p[1] <= '9' && !escaped)
 617                 return TRUE;
 618             escaped = !escaped;
 619         }
 620         else
 621         {
 622             if ((*p == '*' || *p == '?') && !escaped)
 623                 return TRUE;
 624             escaped = FALSE;
 625         }
 626     }
 627     return FALSE;
 628 }

 664 FileProgressStatus
 665 check_progress_buttons (file_op_context_t * ctx)
     /*  [+665 src/filemanager/filegui.c] */
 666 {
 667     int c;
 668     Gpm_Event event;
 669     file_op_context_ui_t *ui;
 670 
 671     if (ctx == NULL || ctx->ui == NULL)
 672         return FILE_CONT;
 673 
 674     ui = ctx->ui;
 675 
 676   get_event:
 677     event.x = -1;               /* Don't show the GPM cursor */
 678     c = tty_get_event (&event, FALSE, ctx->suspended);
 679     if (c == EV_NONE)
 680         return FILE_CONT;
 681 
 682     /* Reinitialize to avoid old values after events other than selecting a button */
 683     ui->op_dlg->ret_value = FILE_CONT;
 684 
 685     dlg_process_event (ui->op_dlg, c, &event);
 686     switch (ui->op_dlg->ret_value)
 687     {
 688     case FILE_SKIP:
 689         if (ctx->suspended)
 690         {
 691             /* redraw dialog in case of Skip after Suspend */
 692             place_progress_buttons (ui->op_dlg, FALSE);
 693             dlg_redraw (ui->op_dlg);
 694         }
 695         ctx->suspended = FALSE;
 696         return FILE_SKIP;
 697     case B_CANCEL:
 698     case FILE_ABORT:
 699         ctx->suspended = FALSE;
 700         return FILE_ABORT;
 701     case FILE_SUSPEND:
 702         ctx->suspended = !ctx->suspended;
 703         place_progress_buttons (ui->op_dlg, ctx->suspended);
 704         dlg_redraw (ui->op_dlg);
 705         /* fallthrough */
 706     default:
 707         if (ctx->suspended)
 708             goto get_event;
 709         return FILE_CONT;
 710     }
 711 }
