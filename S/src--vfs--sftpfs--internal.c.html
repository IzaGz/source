  99 int
 100 sftpfs_waitsocket (sftpfs_super_data_t * super_data, GError ** mcerror)
     /*  [+100 src/vfs/sftpfs/internal.c] */
 101 {
 102     struct timeval timeout = { 10, 0 };
 103     fd_set fd;
 104     fd_set *writefd = NULL;
 105     fd_set *readfd = NULL;
 106     int dir;
 107 
 108     mc_return_val_if_error (mcerror, -1);
 109 
 110     FD_ZERO (&fd);
 111     FD_SET (super_data->socket_handle, &fd);
 112 
 113     /* now make sure we wait in the correct direction */
 114     dir = libssh2_session_block_directions (super_data->session);
 115 
 116     if ((dir & LIBSSH2_SESSION_BLOCK_INBOUND) != 0)
 117         readfd = &fd;
 118 
 119     if ((dir & LIBSSH2_SESSION_BLOCK_OUTBOUND) != 0)
 120         writefd = &fd;
 121 
 122     return select (super_data->socket_handle + 1, readfd, writefd, NULL, &timeout);
 123 }

 149 sftpfs_lstat (const vfs_path_t * vpath, struct stat *buf, GError ** mcerror)
     /*  [+149 src/vfs/sftpfs/internal.c] */
 150 {
 151     struct vfs_s_super *super;
 152     sftpfs_super_data_t *super_data;
 153     LIBSSH2_SFTP_ATTRIBUTES attrs;
 154     int res;
 155     const vfs_path_element_t *path_element;
 156 
 157     mc_return_val_if_error (mcerror, -1);
 158 
 159     path_element = vfs_path_get_by_index (vpath, -1);
 160 
 161     if (vfs_s_get_path (vpath, &super, 0) == NULL)
 162         return -1;
 163 
 164     if (super == NULL)
 165         return -1;
 166 
 167     super_data = (sftpfs_super_data_t *) super->data;
 168     if (super_data->sftp_session == NULL)
 169         return -1;
 170 
 171     do
 172     {
 173         const char *fixfname;
 174         unsigned int fixfname_len = 0;
 175 
 176         fixfname = sftpfs_fix_filename (path_element->path, &fixfname_len);
 177 
 178         res =
 179             libssh2_sftp_stat_ex (super_data->sftp_session, fixfname, fixfname_len,
 180                                   LIBSSH2_SFTP_LSTAT, &attrs);
 181         if (res >= 0)
 182             break;
 183 
 184         if (res != LIBSSH2_ERROR_EAGAIN)
 185         {
 186             sftpfs_ssherror_to_gliberror (super_data, res, mcerror);
 187             return -1;
 188         }
 189 
 190         sftpfs_waitsocket (super_data, mcerror);
 191         mc_return_val_if_error (mcerror, -1);
 192     }
 193     while (res == LIBSSH2_ERROR_EAGAIN);
 194 
 195     if ((attrs.flags & LIBSSH2_SFTP_ATTR_UIDGID) != 0)
 196     {
 197         buf->st_uid = attrs.uid;
 198         buf->st_gid = attrs.gid;
 199     }
 200 
 201     if ((attrs.flags & LIBSSH2_SFTP_ATTR_ACMODTIME) != 0)
 202     {
 203         buf->st_atime = attrs.atime;
 204         buf->st_mtime = attrs.mtime;
 205         buf->st_ctime = attrs.mtime;
 206     }
 207 
 208     if ((attrs.flags & LIBSSH2_SFTP_ATTR_SIZE) != 0)
 209     {
 210         buf->st_size = attrs.filesize;
 211         sftpfs_blksize (buf);
 212     }
 213 
 214     if ((attrs.flags & LIBSSH2_SFTP_ATTR_PERMISSIONS) != 0)
 215         buf->st_mode = attrs.permissions;
 216 
 217     return 0;
 218 }
