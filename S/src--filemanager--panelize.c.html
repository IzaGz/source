 222 static void
 223 add2panelize (char *label, char *command)
     /*  [+223 src/filemanager/panelize.c] */
 224 {
 225     struct panelize *current, *old;
 226 
 227     old = NULL;
 228     current = panelize;
 229     while (current && strcmp (current->label, label) <= 0)
 230     {
 231         old = current;
 232         current = current->next;
 233     }
 234 
 235     if (old == NULL)
 236     {
 237         panelize = g_new (struct panelize, 1);
 238         panelize->label = label;
 239         panelize->command = command;
 240         panelize->next = current;
 241     }
 242     else
 243     {
 244         struct panelize *new;
 245         new = g_new (struct panelize, 1);
 246         new->label = label;
 247         new->command = command;
 248         old->next = new;
 249         new->next = current;
 250     }
 251 }

 274 static void
 275 remove_from_panelize (struct panelize *entry)
     /*  [+275 src/filemanager/panelize.c] */
 276 {
 277     if (strcmp (entry->label, _("Other command")) != 0)
 278     {
 279         if (entry == panelize)
 280         {
 281             panelize = panelize->next;
 282         }
 283         else
 284         {
 285             struct panelize *current = panelize;
 286             while (current && current->next != entry)
 287                 current = current->next;
 288             if (current)
 289             {
 290                 current->next = entry->next;
 291             }
 292         }
 293 
 294         g_free (entry->label);
 295         g_free (entry->command);
 296         g_free (entry);
 297     }
 298 }

 302 static void
 303 do_external_panelize (char *command)
     /*  [+303 src/filemanager/panelize.c] */
 304 {
 305     int link_to_dir, stale_link;
 306     struct stat st;
 307     dir_list *list = &current_panel->dir;
 308     char line[MC_MAXPATHLEN];
 309     char *name;
 310     FILE *external;
 311 
 312     open_error_pipe ();
 313     external = popen (command, "r");
 314     if (!external)
 315     {
 316         close_error_pipe (D_ERROR, _("Cannot invoke command."));
 317         return;
 318     }
 319     /* Clear the counters and the directory list */
 320     panel_clean_dir (current_panel);
 321 
 322     panelize_change_root (current_panel->cwd_vpath);
 323 
 324     dir_list_init (list);
 325 
 326     while (TRUE)
 327     {
 328         clearerr (external);
 329         if (fgets (line, sizeof (line), external) == NULL)
 330         {
 331             if (ferror (external) && errno == EINTR)
 332                 continue;
 333             else
 334                 break;
 335         }
 336         if (line[strlen (line) - 1] == '\n')
 337             line[strlen (line) - 1] = 0;
 338         if (strlen (line) < 1)
 339             continue;
 340         if (line[0] == '.' && IS_PATH_SEP (line[1]))
 341             name = line + 2;
 342         else
 343             name = line;
 344 
 345         if (!handle_path (name, &st, &link_to_dir, &stale_link))
 346             continue;
 347 
 348         if (!dir_list_append (list, name, &st, link_to_dir != 0, stale_link != 0))
 349             break;
 350 
 351         file_mark (current_panel, list->len - 1, 0);
 352 
 353         if ((list->len & 31) == 0)
 354             rotate_dash (TRUE);
 355     }
 356 
 357     current_panel->is_panelized = TRUE;
 358 
 359     if (list->len == 0)
 360         dir_list_init (list);
 361     else if (IS_PATH_SEP (list->list[0].fname[0]))
 362     {
 363         vfs_path_t *vpath_root;
 364         int ret;
 365 
 366         vpath_root = vfs_path_from_str (PATH_SEP_STR);
 367         panel_set_cwd (current_panel, vpath_root);
 368         ret = mc_chdir (vpath_root);
 369         vfs_path_free (vpath_root);
 370 
 371         (void) ret;
 372     }
 373 
 374     if (pclose (external) < 0)
 375         message (D_NORMAL, _("External panelize"), _("Pipe close failed"));
 376     close_error_pipe (D_NORMAL, NULL);
 377     try_to_select (current_panel, NULL);
 378     panel_re_sort (current_panel);
 379     rotate_dash (FALSE);
 380 }

 384 static void
 385 do_panelize_cd (WPanel * panel)
     /*  [+385 src/filemanager/panelize.c] */
 386 {
 387     int i;
 388     dir_list *list;
 389     gboolean panelized_same;
 390 
 391     dir_list_clean (&panel->dir);
 392     if (panelized_panel.root_vpath == NULL)
 393         panelize_change_root (current_panel->cwd_vpath);
 394 
 395     if (panelized_panel.list.len < 1)
 396         dir_list_init (&panelized_panel.list);
 397     else if (panelized_panel.list.len > panel->dir.size)
 398         dir_list_grow (&panel->dir, panelized_panel.list.len - panel->dir.size);
 399 
 400     list = &panel->dir;
 401     list->len = panelized_panel.list.len;
 402     panel->is_panelized = TRUE;
 403 
 404     panelized_same = vfs_path_equal (panelized_panel.root_vpath, panel->cwd_vpath);
 405 
 406     for (i = 0; i < panelized_panel.list.len; i++)
 407     {
 408         if (panelized_same || DIR_IS_DOTDOT (panelized_panel.list.list[i].fname))
 409         {
 410             list->list[i].fnamelen = panelized_panel.list.list[i].fnamelen;
 411             list->list[i].fname = g_strndup (panelized_panel.list.list[i].fname,
 412                                              panelized_panel.list.list[i].fnamelen);
 413         }
 414         else
 415         {
 416             vfs_path_t *tmp_vpath;
 417             const char *fname;
 418 
 419             tmp_vpath =
 420                 vfs_path_append_new (panelized_panel.root_vpath, panelized_panel.list.list[i].fname,
 421                                      (char *) NULL);
 422             fname = vfs_path_as_str (tmp_vpath);
 423             list->list[i].fnamelen = strlen (fname);
 424             list->list[i].fname = g_strndup (fname, list->list[i].fnamelen);
 425             vfs_path_free (tmp_vpath);
 426         }
 427         list->list[i].f.link_to_dir = panelized_panel.list.list[i].f.link_to_dir;
 428         list->list[i].f.stale_link = panelized_panel.list.list[i].f.stale_link;
 429         list->list[i].f.dir_size_computed = panelized_panel.list.list[i].f.dir_size_computed;
 430         list->list[i].f.marked = panelized_panel.list.list[i].f.marked;
 431         list->list[i].st = panelized_panel.list.list[i].st;
 432         list->list[i].sort_key = panelized_panel.list.list[i].sort_key;
 433         list->list[i].second_sort_key = panelized_panel.list.list[i].second_sort_key;
 434     }
 435     try_to_select (panel, NULL);
 436 }

 459     dir_list *list = &panel->dir;

 500 void
 501 external_panelize (void)
     /*  [+501 src/filemanager/panelize.c] */
 502 {
 503     if (!vfs_current_is_local ())
 504     {
 505         message (D_ERROR, MSG_ERROR, _("Cannot run external panelize in a non-local directory"));
 506         return;
 507     }
 508 
 509     init_panelize ();
 510 
 511     /* display file info */
 512     tty_setcolor (SELECTED_COLOR);
 513 
 514     switch (dlg_run (panelize_dlg))
 515     {
 516     case B_CANCEL:
 517         break;
 518 
 519     case B_ADD:
 520         add2panelize_cmd ();
 521         break;
 522 
 523     case B_REMOVE:
 524         {
 525             struct panelize *entry;
 526 
 527             listbox_get_current (l_panelize, NULL, (void **) &entry);
 528             remove_from_panelize (entry);
 529             break;
 530         }
 531 
 532     case B_ENTER:
 533         if (!input_is_empty (pname))
 534         {
 535             char *cmd;
 536 
 537             cmd = g_strdup (pname->buffer);
 538             dlg_destroy (panelize_dlg);
 539             do_external_panelize (cmd);
 540             g_free (cmd);
 541             repaint_screen ();
 542             return;
 543         }
 544         break;
 545 
 546     default:
 547         break;
 548     }
 549 
 550     panelize_done ();
 551 }

 555 void
 556 load_panelize (void)
     /*  [+556 src/filemanager/panelize.c] */
 557 {
 558     char **keys;
 559 
 560     keys = mc_config_get_keys (mc_global.main_config, panelize_section, NULL);
 561 
 562     add2panelize (g_strdup (_("Other command")), g_strdup (""));
 563 
 564     if (*keys == NULL)
 565     {
 566         add2panelize (g_strdup (_("Modified git files")), g_strdup ("git ls-files --modified"));
 567         add2panelize (g_strdup (_("Find rejects after patching")),
 568                       g_strdup ("find . -name \\*.rej -print"));
 569         add2panelize (g_strdup (_("Find *.orig after patching")),
 570                       g_strdup ("find . -name \\*.orig -print"));
 571         add2panelize (g_strdup (_("Find SUID and SGID programs")),
 572                       g_strdup
 573                       ("find . \\( \\( -perm -04000 -a -perm /011 \\) -o \\( -perm -02000 -a -perm /01 \\) \\) -print"));
 574     }
 575     else
 576     {
 577         GIConv conv;
 578         char **profile_keys;
 579 
 580         conv = str_crt_conv_from ("UTF-8");
 581 
 582         for (profile_keys = keys; *profile_keys != NULL; profile_keys++)
 583         {
 584             GString *buffer;
 585 
 586             if (mc_global.utf8_display || conv == INVALID_CONV)
 587                 buffer = g_string_new (*profile_keys);
 588             else
 589             {
 590                 buffer = g_string_new ("");
 591                 if (str_convert (conv, *profile_keys, buffer) == ESTR_FAILURE)
 592                     g_string_assign (buffer, *profile_keys);
 593             }
 594 
 595             add2panelize (g_string_free (buffer, FALSE),
 596                           mc_config_get_string (mc_global.main_config, panelize_section,
 597                                                 *profile_keys, ""));
 598         }
 599 
 600         str_close_conv (conv);
 601     }
 602 
 603     g_strfreev (keys);
 604 }
