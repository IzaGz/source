 150 static void
 151 undelfs_get_path (const vfs_path_t * vpath, char **fsname, char **file)
     /*  [+151 src/vfs/undelfs/undelfs.c] */
 152 {
 153     const char *p, *dirname;
 154     const vfs_path_element_t *path_element;
 155 
 156     path_element = vfs_path_get_by_index (vpath, -1);
 157 
 158     /* To look like filesystem, we have virtual directories
 159        undel://XXX, which have no subdirectories. XXX is replaced with
 160        hda5, sdb8 etc, which is assumed to live under /dev. 
 161        -- pavel@ucw.cz */
 162 
 163     dirname = path_element->path;
 164 
 165     *fsname = NULL;
 166 
 167     if (strncmp (dirname, "undel://", 8) != 0)
 168         return;
 169 
 170     dirname += 8;
 171 
 172     /* Since we don't allow subdirectories, it's easy to get a filename,
 173      * just scan backwards for a slash */
 174     if (*dirname == 0)
 175         return;
 176 
 177     p = dirname + strlen (dirname);
 178 #if 0
 179     /* Strip trailing ./
 180      */
 181     if (p - dirname > 2 && IS_PATH_SEP (p[-1]) && p[-2] == '.')
 182         *(p = p - 2) = 0;
 183 #endif
 184 
 185     while (p > dirname)
 186     {
 187         if (IS_PATH_SEP (*p))
 188         {
 189             char *tmp;
 190 
 191             *file = g_strdup (p + 1);
 192             tmp = g_strndup (dirname, p - dirname);
 193             *fsname = g_strconcat ("/dev/", tmp, (char *) NULL);
 194             g_free (tmp);
 195             return;
 196         }
 197         p--;
 198     }
 199     *file = g_strdup ("");
 200     *fsname = g_strconcat ("/dev/", dirname, (char *) NULL);
 201 }

 230 static int
 231 undelfs_loaddel (void)
     /*  [+231 src/vfs/undelfs/undelfs.c] */
 232 {
 233     int retval, count;
 234     ext2_ino_t ino;
 235     struct ext2_inode inode;
 236     ext2_inode_scan scan;
 237 
 238     max_delarray = 100;
 239     num_delarray = 0;
 240     delarray = g_try_malloc (sizeof (struct deleted_info) * max_delarray);
 241     if (!delarray)
 242     {
 243         message (D_ERROR, undelfserr, _("not enough memory"));
 244         return 0;
 245     }
 246     block_buf = g_try_malloc (fs->blocksize * 3);
 247     if (!block_buf)
 248     {
 249         message (D_ERROR, undelfserr, _("while allocating block buffer"));
 250         goto free_delarray;
 251     }
 252     retval = ext2fs_open_inode_scan (fs, 0, &scan);
 253     if (retval != 0)
 254     {
 255         message (D_ERROR, undelfserr, _("open_inode_scan: %d"), retval);
 256         goto free_block_buf;
 257     }
 258     retval = ext2fs_get_next_inode (scan, &ino, &inode);
 259     if (retval != 0)
 260     {
 261         message (D_ERROR, undelfserr, _("while starting inode scan %d"), retval);
 262         goto error_out;
 263     }
 264     count = 0;
 265     while (ino)
 266     {
 267         if ((count++ % 1024) == 0)
 268             vfs_print_message (_("undelfs: loading deleted files information %d inodes"), count);
 269         if (inode.i_dtime == 0)
 270             goto next;
 271 
 272         if (S_ISDIR (inode.i_mode))
 273             goto next;
 274 
 275         lsd.inode = ino;
 276         lsd.num_blocks = 0;
 277         lsd.free_blocks = 0;
 278         lsd.bad_blocks = 0;
 279 
 280         retval = ext2fs_block_iterate (fs, ino, 0, block_buf, undelfs_lsdel_proc, &lsd);
 281         if (retval)
 282         {
 283             message (D_ERROR, undelfserr, _("while calling ext2_block_iterate %d"), retval);
 284             goto next;
 285         }
 286         if (lsd.free_blocks && !lsd.bad_blocks)
 287         {
 288             if (num_delarray >= max_delarray)
 289             {
 290                 struct deleted_info *delarray_new = g_try_realloc (delarray,
 291                                                                    sizeof (struct deleted_info) *
 292                                                                    (max_delarray + 50));
 293                 if (!delarray_new)
 294                 {
 295                     message (D_ERROR, undelfserr, _("no more memory while reallocating array"));
 296                     goto error_out;
 297                 }
 298                 delarray = delarray_new;
 299                 max_delarray += 50;
 300             }
 301 
 302             delarray[num_delarray].ino = ino;
 303             delarray[num_delarray].mode = inode.i_mode;
 304             delarray[num_delarray].uid = inode.i_uid;
 305             delarray[num_delarray].gid = inode.i_gid;
 306             delarray[num_delarray].size = inode.i_size;
 307             delarray[num_delarray].dtime = inode.i_dtime;
 308             delarray[num_delarray].num_blocks = lsd.num_blocks;
 309             delarray[num_delarray].free_blocks = lsd.free_blocks;
 310             num_delarray++;
 311         }
 312 
 313       next:
 314         retval = ext2fs_get_next_inode (scan, &ino, &inode);
 315         if (retval)
 316         {
 317             message (D_ERROR, undelfserr, _("while doing inode scan %d"), retval);
 318             goto error_out;
 319         }
 320     }
 321     readdir_ptr = READDIR_PTR_INIT;
 322     ext2fs_close_inode_scan (scan);
 323     return 1;
 324 
 325   error_out:
 326     ext2fs_close_inode_scan (scan);
 327   free_block_buf:
 328     MC_PTR_FREE (block_buf);
 329   free_delarray:
 330     MC_PTR_FREE (delarray);
 331     return 0;
 332 }

 336 static void *
 337 undelfs_opendir (const vfs_path_t * vpath)
     /*  [+337 src/vfs/undelfs/undelfs.c] */
 338 {
 339     char *file, *f = NULL;
 340     const vfs_path_element_t *path_element;
 341 
 342     path_element = vfs_path_get_by_index (vpath, -1);
 343     undelfs_get_path (vpath, &file, &f);
 344     if (file == NULL)
 345     {
 346         g_free (f);
 347         return 0;
 348     }
 349 
 350     /* We don't use the file name */
 351     g_free (f);
 352 
 353     if (!ext2_fname || strcmp (ext2_fname, file))
 354     {
 355         undelfs_shutdown ();
 356         ext2_fname = file;
 357     }
 358     else
 359     {
 360         /* To avoid expensive re-scannings */
 361         readdir_ptr = READDIR_PTR_INIT;
 362         g_free (file);
 363         return fs;
 364     }
 365 
 366     if (ext2fs_open (ext2_fname, 0, 0, 0, unix_io_manager, &fs))
 367     {
 368         message (D_ERROR, undelfserr, _("Cannot open file %s"), ext2_fname);
 369         return 0;
 370     }
 371     vfs_print_message ("%s", _("undelfs: reading inode bitmap..."));
 372     if (ext2fs_read_inode_bitmap (fs))
 373     {
 374         message (D_ERROR, undelfserr, _("Cannot load inode bitmap from:\n%s"), ext2_fname);
 375         goto quit_opendir;
 376     }
 377     vfs_print_message ("%s", _("undelfs: reading block bitmap..."));
 378     if (ext2fs_read_block_bitmap (fs))
 379     {
 380         message (D_ERROR, undelfserr, _("Cannot load block bitmap from:\n%s"), ext2_fname);
 381         goto quit_opendir;
 382     }
 383     /* Now load the deleted information */
 384     if (!undelfs_loaddel ())
 385         goto quit_opendir;
 386     vfs_print_message (_("%s: done."), path_element->class->name);
 387     return fs;
 388   quit_opendir:
 389     vfs_print_message (_("%s: failure"), path_element->class->name);
 390     ext2fs_close (fs);
 391     fs = NULL;
 392     return 0;
 393 }

 432 static void *
 433 undelfs_open (const vfs_path_t * vpath, int flags, mode_t mode)
     /*  [+433 src/vfs/undelfs/undelfs.c] */
 434 {
 435     char *file, *f = NULL;
 436     ext2_ino_t inode, i;
 437     undelfs_file *p = NULL;
 438     (void) flags;
 439     (void) mode;
 440 
 441     /* Only allow reads on this file system */
 442     undelfs_get_path (vpath, &file, &f);
 443     if (file == NULL)
 444     {
 445         g_free (f);
 446         return 0;
 447     }
 448 
 449     if (!ext2_fname || strcmp (ext2_fname, file))
 450     {
 451         message (D_ERROR, undelfserr, _("You have to chdir to extract files first"));
 452         g_free (file);
 453         g_free (f);
 454         return 0;
 455     }
 456     inode = atol (f);
 457 
 458     /* Search the file into delarray */
 459     for (i = 0; i < (ext2_ino_t) num_delarray; i++)
 460     {
 461         if (inode != delarray[i].ino)
 462             continue;
 463 
 464         /* Found: setup all the structures needed by read */
 465         p = (undelfs_file *) g_try_malloc (((gsize) sizeof (undelfs_file)));
 466         if (!p)
 467         {
 468             g_free (file);
 469             g_free (f);
 470             return 0;
 471         }
 472         p->buf = g_try_malloc (fs->blocksize);
 473         if (!p->buf)
 474         {
 475             g_free (p);
 476             g_free (file);
 477             g_free (f);
 478             return 0;
 479         }
 480         p->inode = inode;
 481         p->finished = FALSE;
 482         p->f_index = i;
 483         p->error_code = 0;
 484         p->pos = 0;
 485         p->size = delarray[i].size;
 486     }
 487     g_free (file);
 488     g_free (f);
 489     undelfs_usage++;
 490     return p;
 491 }

 507 static int
 508 undelfs_dump_read (ext2_filsys param_fs, blk_t * blocknr, int blockcnt, void *private)
     /*  [+508 src/vfs/undelfs/undelfs.c] */
 509 {
 510     int copy_count;
 511     undelfs_file *p = (undelfs_file *) private;
 512 
 513     if (blockcnt < 0)
 514         return 0;
 515 
 516     if (*blocknr)
 517     {
 518         p->error_code = io_channel_read_blk (param_fs->io, *blocknr, 1, p->buf);
 519         if (p->error_code)
 520             return BLOCK_ABORT;
 521     }
 522     else
 523         memset (p->buf, 0, param_fs->blocksize);
 524 
 525     if (p->pos + (off_t) p->count < p->current)
 526     {
 527         p->finished = TRUE;
 528         return BLOCK_ABORT;
 529     }
 530     if (p->pos > p->current + param_fs->blocksize)
 531     {
 532         p->current += param_fs->blocksize;
 533         return 0;               /* we have not arrived yet */
 534     }
 535 
 536     /* Now, we know we have to extract some data */
 537     if (p->pos >= p->current)
 538     {
 539 
 540         /* First case: starting pointer inside this block */
 541         if (p->pos + (off_t) p->count <= p->current + param_fs->blocksize)
 542         {
 543             /* Fully contained */
 544             copy_count = p->count;
 545             p->finished = (p->count != 0);
 546         }
 547         else
 548         {
 549             /* Still some more data */
 550             copy_count = param_fs->blocksize - (p->pos - p->current);
 551         }
 552         memcpy (p->dest_buffer, p->buf + (p->pos - p->current), copy_count);
 553     }
 554     else
 555     {
 556         /* Second case: we already have passed p->pos */
 557         if (p->pos + (off_t) p->count < p->current + param_fs->blocksize)
 558         {
 559             copy_count = (p->pos + p->count) - p->current;
 560             p->finished = (p->count != 0);
 561         }
 562         else
 563         {
 564             copy_count = param_fs->blocksize;
 565         }
 566         memcpy (p->dest_buffer, p->buf, copy_count);
 567     }
 568     p->dest_buffer += copy_count;
 569     p->current += param_fs->blocksize;
 570     if (p->finished)
 571     {
 572         return BLOCK_ABORT;
 573     }
 574     return 0;
 575 }

 579 static ssize_t
 580 undelfs_read (void *vfs_info, char *buffer, size_t count)
     /*  [+580 src/vfs/undelfs/undelfs.c] */
 581 {
 582     undelfs_file *p = vfs_info;
 583     int retval;
 584 
 585     p->dest_buffer = buffer;
 586     p->current = 0;
 587     p->finished = FALSE;
 588     p->count = count;
 589 
 590     if (p->pos + (off_t) p->count > p->size)
 591     {
 592         p->count = p->size - p->pos;
 593     }
 594     retval = ext2fs_block_iterate (fs, p->inode, 0, NULL, undelfs_dump_read, p);
 595     if (retval)
 596     {
 597         message (D_ERROR, undelfserr, _("while iterating over blocks"));
 598         return -1;
 599     }
 600     if (p->error_code && !p->finished)
 601         return 0;
 602     p->pos = p->pos + (p->dest_buffer - buffer);
 603     return p->dest_buffer - buffer;
 604 }

 649     if (file == NULL)
 650     {
 651         g_free (f);
 652         return 0;
 653     }
 654 
 655     /* When called from save_cwd_stats we get an incorrect file and f here:
 656        e.g. incorrect                         correct
 657        path = "undel:/dev/sda1"          path="undel:/dev/sda1/401:1"
 658        file = "/dev"                     file="/dev/sda1"

 665         return -1;
 666     }

 674     }

 676     g_free (file);
 677     g_free (f);
 678

 697 static int

     /*  [+698 src/vfs/undelfs/undelfs.c] */
 699 {
 700     char *file, *f = NULL;
 701     int fd;
 702 
 703     undelfs_get_path (vpath, &file, &f);
 704     if (file == NULL)
 705     {
 706         g_free (f);
 707         return (-1);
 708     }
 709 
 710     /* We may use access because ext2 file systems are local */
 711     /* this could be fixed by making an ext2fs io manager to use */
 712     /* our vfs, but that is left as an exercise for the reader */
 713     fd = open (file, O_RDONLY);
 714     if (fd == -1)
 715     {
 716         message (D_ERROR, undelfserr, _("Cannot open file \"%s\""), file);
 717         g_free (f);
 718         g_free (file);
 719         return -1;
 720     }
 721     close (fd);
 722     g_free (f);
 723     g_free (file);
 724     return 0;
 725 }
