 234 /**
 235  * Quick and dirty octal conversion.
 236  *
 237  * Result is -1 if the field is invalid (all blank, or nonoctal).
 238  */
 239 static long
 240 tar_from_oct (int digs, char *where)
     /*  [+240 src/vfs/tar/tar.c] */
 241 {
 242     long value;
 243 
 244     while (isspace ((unsigned char) *where))
 245     {                           /* Skip spaces */
 246         where++;
 247         if (--digs <= 0)
 248             return -1;          /* All blank field */
 249     }
 250     value = 0;
 251     while (digs > 0 && isodigit (*where))
 252     {                           /* Scan till nonoctal */
 253         value = (value << 3) | (*where++ - '0');
 254         --digs;
 255     }
 256 
 257     if (digs > 0 && *where && !isspace ((unsigned char) *where))
 258         return -1;              /* Ended on non-space/nul */
 259 
 260     return value;
 261 }

 283 static int
 284 tar_open_archive_int (struct vfs_class *me, const vfs_path_t * vpath, struct vfs_s_super *archive)
     /*  [+284 src/vfs/tar/tar.c] */
 285 {
 286     int result, type;
 287     tar_super_data_t *arch;
 288     mode_t mode;
 289     struct vfs_s_inode *root;
 290 
 291     result = mc_open (vpath, O_RDONLY);
 292     if (result == -1)
 293     {
 294         message (D_ERROR, MSG_ERROR, _("Cannot open tar archive\n%s"), vfs_path_as_str (vpath));
 295         ERRNOR (ENOENT, -1);
 296     }
 297 
 298     archive->name = g_strdup (vfs_path_as_str (vpath));
 299     archive->data = g_new (tar_super_data_t, 1);
 300     arch = (tar_super_data_t *) archive->data;
 301     mc_stat (vpath, &arch->st);
 302     arch->fd = -1;
 303     arch->type = TAR_UNKNOWN;
 304 
 305     /* Find out the method to handle this tar file */
 306     type = get_compression_type (result, archive->name);
 307     if (type == COMPRESSION_NONE)
 308         mc_lseek (result, 0, SEEK_SET);
 309     else
 310     {
 311         char *s;
 312         vfs_path_t *tmp_vpath;
 313 
 314         mc_close (result);
 315         s = g_strconcat (archive->name, decompress_extension (type), (char *) NULL);
 316         tmp_vpath = vfs_path_from_str_flags (s, VPF_NO_CANON);
 317         result = mc_open (tmp_vpath, O_RDONLY);
 318         vfs_path_free (tmp_vpath);
 319         if (result == -1)
 320             message (D_ERROR, MSG_ERROR, _("Cannot open tar archive\n%s"), s);
 321         g_free (s);
 322         if (result == -1)
 323         {
 324             MC_PTR_FREE (archive->name);
 325             ERRNOR (ENOENT, -1);
 326         }
 327     }
 328 
 329     arch->fd = result;
 330     mode = arch->st.st_mode & 07777;
 331     if (mode & 0400)
 332         mode |= 0100;
 333     if (mode & 0040)
 334         mode |= 0010;
 335     if (mode & 0004)
 336         mode |= 0001;
 337     mode |= S_IFDIR;
 338 
 339     root = vfs_s_new_inode (me, archive, &arch->st);
 340     root->st.st_mode = mode;
 341     root->data_offset = -1;
 342     root->st.st_nlink++;
 343     root->st.st_dev = MEDATA->rdev++;
 344 
 345     archive->root = root;
 346 
 347     return result;
 348 }

 379 static void
 380 tar_fill_stat (struct vfs_s_super *archive, struct stat *st, union record *header, size_t h_size)
     /*  [+380 src/vfs/tar/tar.c] */
 381 {
 382     tar_super_data_t *arch = (tar_super_data_t *) archive->data;
 383 
 384     st->st_mode = tar_from_oct (8, header->header.mode);
 385 
 386     /* Adjust st->st_mode because there are tar-files with
 387      * linkflag==LF_SYMLINK and S_ISLNK(mod)==0. I don't 
 388      * know about the other modes but I think I cause no new
 389      * problem when I adjust them, too. -- Norbert.
 390      */
 391     if (header->header.linkflag == LF_DIR)
 392         st->st_mode |= S_IFDIR;
 393     else if (header->header.linkflag == LF_SYMLINK)
 394         st->st_mode |= S_IFLNK;
 395     else if (header->header.linkflag == LF_CHR)
 396         st->st_mode |= S_IFCHR;
 397     else if (header->header.linkflag == LF_BLK)
 398         st->st_mode |= S_IFBLK;
 399     else if (header->header.linkflag == LF_FIFO)
 400         st->st_mode |= S_IFIFO;
 401     else
 402         st->st_mode |= S_IFREG;
 403 
 404     st->st_dev = 0;
 405 #ifdef HAVE_STRUCT_STAT_ST_RDEV
 406     st->st_rdev = 0;
 407 #endif
 408 
 409     switch (arch->type)
 410     {
 411     case TAR_USTAR:
 412     case TAR_POSIX:
 413     case TAR_GNU:
 414         /* *INDENT-OFF* */
 415         st->st_uid = *header->header.uname
 416             ? vfs_finduid (header->header.uname)
 417             : tar_from_oct (8, header->header.uid);
 418         st->st_gid = *header->header.gname
 419             ? vfs_findgid (header->header.gname)
 420             : tar_from_oct (8,header->header.gid);
 421         /* *INDENT-ON* */
 422 
 423         switch (header->header.linkflag)
 424         {
 425         case LF_BLK:
 426         case LF_CHR:
 427 #ifdef HAVE_STRUCT_STAT_ST_RDEV
 428             st->st_rdev =
 429                 makedev (tar_from_oct (8, header->header.devmajor),
 430                          tar_from_oct (8, header->header.devminor));
 431 #endif
 432             break;
 433         default:
 434             break;
 435         }
 436         break;
 437 
 438     default:
 439         st->st_uid = tar_from_oct (8, header->header.uid);
 440         st->st_gid = tar_from_oct (8, header->header.gid);
 441         break;
 442     }
 443 
 444     st->st_size = h_size;
 445     st->st_mtime = tar_from_oct (1 + 12, header->header.mtime);
 446     st->st_atime = 0;
 447     st->st_ctime = 0;
 448     if (arch->type == TAR_GNU)
 449     {
 450         st->st_atime = tar_from_oct (1 + 12, header->header.unused.oldgnu.atime);
 451         st->st_ctime = tar_from_oct (1 + 12, header->header.unused.oldgnu.ctime);
 452     }
 453 
 454 #ifdef HAVE_STRUCT_STAT_ST_BLKSIZE
 455     st->st_blksize = 8 * 1024;  /* FIXME */
 456 #endif
 457     vfs_adjust_stat (st);
 458 }

 461 /**
 462  * Return 1 for success, 0 if the checksum is bad, EOF on eof,
 463  * 2 for a record full of zeros (EOF marker).
 464  *
 465  */
 466 static ReadStatus
 467 tar_read_header (struct vfs_class *me, struct vfs_s_super *archive, int tard, size_t * h_size)
     /*  [+467 src/vfs/tar/tar.c] */
 468 {
 469     tar_super_data_t *arch = (tar_super_data_t *) archive->data;
 470 
 471     int i;
 472     long sum, signed_sum, recsum;
 473     char *p;
 474     union record *header;
 475     static char *next_long_name = NULL, *next_long_link = NULL;
 476 
 477   recurse:
 478 
 479     header = tar_get_next_record (archive, tard);
 480     if (NULL == header)
 481         return STATUS_EOF;
 482 
 483     recsum = tar_from_oct (8, header->header.chksum);
 484 
 485     sum = 0;
 486     signed_sum = 0;
 487     p = header->charptr;
 488     for (i = sizeof (*header); --i >= 0;)
 489     {
 490         /*
 491          * We can't use unsigned char here because of old compilers,
 492          * e.g. V7.
 493          */
 494         signed_sum += *p;
 495         sum += 0xFF & *p++;
 496     }
 497 
 498     /* Adjust checksum to count the "chksum" field as blanks. */
 499     for (i = sizeof (header->header.chksum); --i >= 0;)
 500     {
 501         sum -= 0xFF & header->header.chksum[i];
 502         signed_sum -= (char) header->header.chksum[i];
 503     }
 504     sum += ' ' * sizeof header->header.chksum;
 505     signed_sum += ' ' * sizeof header->header.chksum;
 506 
 507     /*
 508      * This is a zeroed record...whole record is 0's except
 509      * for the 8 blanks we faked for the checksum field.
 510      */
 511     if (sum == 8 * ' ')
 512         return STATUS_EOFMARK;
 513 
 514     if (sum != recsum && signed_sum != recsum)
 515         return STATUS_BADCHECKSUM;
 516 
 517     /*
 518      * Try to determine the archive format.
 519      */
 520     if (arch->type == TAR_UNKNOWN)
 521     {
 522         if (strcmp (header->header.magic, TMAGIC) == 0)
 523         {
 524             if (header->header.linkflag == LF_GLOBAL_EXTHDR)
 525                 arch->type = TAR_POSIX;
 526             else
 527                 arch->type = TAR_USTAR;
 528         }
 529         else if (strcmp (header->header.magic, OLDGNU_MAGIC) == 0)
 530             arch->type = TAR_GNU;
 531     }
 532 
 533     /*
 534      * linkflag on BSDI tar (pax) always '\000'
 535      */
 536     if (header->header.linkflag == '\000')
 537     {
 538         size_t len;
 539 
 540         if (header->header.arch_name[NAMSIZ - 1] != '\0')
 541             len = NAMSIZ;
 542         else
 543             len = strlen (header->header.arch_name);
 544 
 545         if (len != 0 && IS_PATH_SEP (header->header.arch_name[len - 1]))
 546             header->header.linkflag = LF_DIR;
 547     }
 548 
 549     /*
 550      * Good record.  Decode file size and return.
 551      */
 552     if (header->header.linkflag == LF_LINK || header->header.linkflag == LF_DIR)
 553         *h_size = 0;            /* Links 0 size on tape */
 554     else
 555         *h_size = tar_from_oct (1 + 12, header->header.size);
 556 
 557     /*
 558      * Skip over directory snapshot info records that
 559      * are stored in incremental tar archives.
 560      */
 561     if (header->header.linkflag == LF_DUMPDIR)
 562     {
 563         if (arch->type == TAR_UNKNOWN)
 564             arch->type = TAR_GNU;
 565         return STATUS_SUCCESS;
 566     }
 567 
 568     /*
 569      * Skip over pax extended header and global extended
 570      * header records.
 571      */
 572     if (header->header.linkflag == LF_EXTHDR || header->header.linkflag == LF_GLOBAL_EXTHDR)
 573     {
 574         if (arch->type == TAR_UNKNOWN)
 575             arch->type = TAR_POSIX;
 576         return STATUS_SUCCESS;
 577     }
 578 
 579     if (header->header.linkflag == LF_LONGNAME || header->header.linkflag == LF_LONGLINK)
 580     {
 581         char **longp;
 582         char *bp, *data;
 583         off_t size;
 584         size_t written;
 585 
 586         if (arch->type == TAR_UNKNOWN)
 587             arch->type = TAR_GNU;
 588 
 589         if (*h_size > MC_MAXPATHLEN)
 590         {
 591             message (D_ERROR, MSG_ERROR, _("Inconsistent tar archive"));
 592             return STATUS_BADCHECKSUM;
 593         }
 594 
 595         longp = ((header->header.linkflag == LF_LONGNAME) ? &next_long_name : &next_long_link);
 596 
 597         g_free (*longp);
 598         bp = *longp = g_malloc (*h_size + 1);
 599 
 600         for (size = *h_size; size > 0; size -= written)
 601         {
 602             data = tar_get_next_record (archive, tard)->charptr;
 603             if (data == NULL)
 604             {
 605                 MC_PTR_FREE (*longp);
 606                 message (D_ERROR, MSG_ERROR, _("Unexpected EOF on archive file"));
 607                 return STATUS_BADCHECKSUM;
 608             }
 609             written = RECORDSIZE;
 610             if ((off_t) written > size)
 611                 written = (size_t) size;
 612 
 613             memcpy (bp, data, written);
 614             bp += written;
 615         }
 616 
 617         if (bp - *longp == MC_MAXPATHLEN && bp[-1] != '\0')
 618         {
 619             MC_PTR_FREE (*longp);
 620             message (D_ERROR, MSG_ERROR, _("Inconsistent tar archive"));
 621             return STATUS_BADCHECKSUM;
 622         }
 623         *bp = 0;
 624         goto recurse;
 625     }
 626     else
 627     {
 628         struct stat st;
 629         struct vfs_s_entry *entry;
 630         struct vfs_s_inode *inode = NULL, *parent;
 631         off_t data_position;
 632         char *q;
 633         size_t len;
 634         char *current_file_name, *current_link_name;
 635 
 636         current_link_name =
 637             (next_long_link ? next_long_link : g_strndup (header->header.arch_linkname, NAMSIZ));
 638         len = strlen (current_link_name);
 639         if (len > 1 && IS_PATH_SEP (current_link_name[len - 1]))
 640             current_link_name[len - 1] = '\0';
 641 
 642         current_file_name = NULL;
 643         switch (arch->type)
 644         {
 645         case TAR_USTAR:
 646         case TAR_POSIX:
 647             /* The ustar archive format supports pathnames of upto 256
 648              * characters in length. This is achieved by concatenating
 649              * the contents of the 'prefix' and 'arch_name' fields like
 650              * this:
 651              *
 652              *   prefix + path_separator + arch_name
 653              *
 654              * If the 'prefix' field contains an empty string i.e. its
 655              * first characters is '\0' the prefix field is ignored.
 656              */
 657             if (header->header.unused.prefix[0] != '\0')
 658             {
 659                 char *temp_name, *temp_prefix;
 660 
 661                 temp_name = g_strndup (header->header.arch_name, NAMSIZ);
 662                 temp_prefix = g_strndup (header->header.unused.prefix, PREFIX_SIZE);
 663                 current_file_name = g_strconcat (temp_prefix, PATH_SEP_STR,
 664                                                  temp_name, (char *) NULL);
 665                 g_free (temp_name);
 666                 g_free (temp_prefix);
 667             }
 668             break;
 669         case TAR_GNU:
 670             if (next_long_name != NULL)

 672             break;
 673         default:
 674             break;
 675         }
 676

 678         {
 679             if (next_long_name != NULL)

 692         {
 693             p = current_file_name;

 695         }
 696         else
 697         {
 698             *(p++) = 0;

 707         }

 713             {
 714                 message (D_ERROR, MSG_ERROR, _("Inconsistent tar archive"));
 715             }
 716             else
 717             {
 718                 entry = vfs_s_new_entry (me, p, inode);
 719                 vfs_s_insert_entry (me, parent, entry);
 720                 g_free (current_link_name);
 721                 goto done;
 722             }
 723         }
 724 
 725         tar_fill_stat (archive, &st, header, *h_size);
 726         if (S_ISDIR (st.st_mode))
 727         {
 728             entry = MEDATA->find_entry (me, parent, p, LINK_NO_FOLLOW, FL_NONE);
 729             if (entry)
 730                 goto done;
 731         }
 732         inode = vfs_s_new_inode (me, archive, &st);
 733 
 734         inode->data_offset = data_position;
 735         if (*current_link_name)
 736         {
 737             inode->linkname = current_link_name;
 738         }
 739         else if (current_link_name != next_long_link)
 740         {
 741             g_free (current_link_name);
 742         }
 743         entry = vfs_s_new_entry (me, p, inode);
 744 
 745         vfs_s_insert_entry (me, parent, entry);
 746         g_free (current_file_name);
 747 
 748       done:
 749         next_long_link = next_long_name = NULL;
 750 
 751         if (arch->type == TAR_GNU && header->header.unused.oldgnu.isextended)
 752         {
 753             while (tar_get_next_record (archive, tard)->ext_hdr.isextended != 0)
 754                 ;
 755 
 756             if (inode != NULL)
 757                 inode->data_offset = current_tar_position;
 758         }
 759         return STATUS_SUCCESS;
 760     }
 761 }

 764 /**
 765  * Main loop for reading an archive.
 766  * Returns 0 on success, -1 on error.
 767  */
 768 static int
 769 tar_open_archive (struct vfs_s_super *archive, const vfs_path_t * vpath,
     /*  [+769 src/vfs/tar/tar.c] */
 770                   const vfs_path_element_t * vpath_element)
 771 {
 772     /* Initial status at start of archive */
 773     ReadStatus status = STATUS_EOFMARK;
 774     int tard;
 775 
 776     current_tar_position = 0;
 777     /* Open for reading */
 778     tard = tar_open_archive_int (vpath_element->class, vpath, archive);
 779     if (tard == -1)
 780         return -1;
 781 
 782     while (TRUE)
 783     {
 784         size_t h_size;
 785         ReadStatus prev_status = status;
 786 
 787         status = tar_read_header (vpath_element->class, archive, tard, &h_size);
 788 
 789         switch (status)
 790         {
 791         case STATUS_SUCCESS:
 792             tar_skip_n_records (archive, tard, (h_size + RECORDSIZE - 1) / RECORDSIZE);
 793             continue;
 794 
 795             /*
 796              * Invalid header:
 797              *
 798              * If the previous header was good, tell them
 799              * that we are skipping bad ones.
 800              */
 801         case STATUS_BADCHECKSUM:
 802             switch (prev_status)
 803             {
 804                 /* Error on first record */
 805             case STATUS_EOFMARK:
 806                 {
 807                     message (D_ERROR, MSG_ERROR, _("%s\ndoesn't look like a tar archive."),
 808                              vfs_path_as_str (vpath));
 809                     /* FALL THRU */
 810 
 811                     /* Error after header rec */
 812                 }
 813             case STATUS_SUCCESS:
 814                 /* Error after error */
 815 
 816             case STATUS_BADCHECKSUM:
 817                 return -1;
 818 
 819             case STATUS_EOF:
 820                 return 0;
 821 
 822             default:
 823                 break;
 824             }
 825 
 826             /* Record of zeroes */
 827         case STATUS_EOFMARK:   /* If error after 0's */
 828             /* FALL THRU */
 829             /* exit from loop */
 830         case STATUS_EOF:       /* End of archive */
 831             break;
 832         default:
 833             break;
 834         }
 835         break;
 836     }
 837     return 0;
 838 }

 882 tar_read (void *fh, char *buffer, size_t count)
     /*  [+882 src/vfs/tar/tar.c] */
 883 {

 889     if (mc_lseek (fd, begin + FH->pos, SEEK_SET) != begin + FH->pos)
 890         ERRNOR (EIO, -1);

 892     count = MIN (count, (size_t) (FH->ino->st.st_size - FH->pos));

 894     res = mc_read (fd, buffer, count);
 895     if (res == -1)
 896         ERRNOR (errno, -1);
 897 
 898     FH->pos += res;
 899     return res;
