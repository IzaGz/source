 225 static char *
 226 exec_make_shell_string (const char *lc_data, const vfs_path_t * filename_vpath)
     /*  [+226 src/filemanager/ext.c] */
 227 {
 228     GString *shell_string;
 229     char lc_prompt[80] = "\0";
 230     gboolean parameter_found = FALSE;
 231     gboolean expand_prefix_found = FALSE;
 232 
 233     shell_string = g_string_new ("");
 234 
 235     for (; *lc_data != '\0' && *lc_data != '\n'; lc_data++)
 236     {
 237         if (parameter_found)
 238         {
 239             if (*lc_data == '}')
 240             {
 241                 char *parameter;
 242 
 243                 parameter_found = FALSE;
 244                 parameter =
 245                     input_dialog (_("Parameter"), lc_prompt, MC_HISTORY_EXT_PARAMETER, "",
 246                                   INPUT_COMPLETE_NONE);
 247                 if (parameter == NULL)
 248                 {
 249                     /* User canceled */
 250                     g_string_free (shell_string, TRUE);
 251                     exec_cleanup_file_name (filename_vpath, FALSE);
 252                     return NULL;
 253                 }
 254                 g_string_append (shell_string, parameter);
 255                 written_nonspace = TRUE;
 256                 g_free (parameter);
 257             }
 258             else
 259             {
 260                 size_t len = strlen (lc_prompt);
 261 
 262                 if (len < sizeof (lc_prompt) - 1)
 263                 {
 264                     lc_prompt[len] = *lc_data;
 265                     lc_prompt[len + 1] = '\0';
 266                 }
 267             }
 268         }
 269         else if (expand_prefix_found)
 270         {
 271             expand_prefix_found = FALSE;
 272             if (*lc_data == '{')
 273                 parameter_found = TRUE;
 274             else
 275             {
 276                 int i;
 277                 char *v;
 278 
 279                 i = check_format_view (lc_data);
 280                 if (i != 0)
 281                 {
 282                     lc_data += i - 1;
 283                     run_view = TRUE;
 284                 }
 285                 else
 286                 {
 287                     i = check_format_cd (lc_data);
 288                     if (i > 0)
 289                     {
 290                         is_cd = TRUE;
 291                         quote_func = fake_name_quote;
 292                         do_local_copy = FALSE;
 293                         pbuffer = buffer;
 294                         lc_data += i - 1;
 295                     }
 296                     else
 297                     {
 298                         i = check_format_var (lc_data, &v);
 299                         if (i > 0 && v != NULL)
 300                         {
 301                             g_string_append (shell_string, v);
 302                             g_free (v);
 303                             lc_data += i;
 304                         }
 305                         else
 306                         {
 307                             char *text;
 308 
 309                             if (*lc_data != 'f')
 310                                 text = expand_format (NULL, *lc_data, !is_cd);
 311                             else
 312                             {
 313                                 text = exec_get_file_name (filename_vpath);
 314                                 if (text == NULL)
 315                                 {
 316                                     g_string_free (shell_string, TRUE);
 317                                     return NULL;
 318                                 }
 319                             }
 320 
 321                             if (!is_cd)
 322                                 g_string_append (shell_string, text);
 323                             else
 324                             {
 325                                 strcpy (pbuffer, text);
 326                                 pbuffer = strchr (pbuffer, 0);
 327                             }
 328 
 329                             g_free (text);
 330                             written_nonspace = TRUE;
 331                         }
 332                     }
 333                 }
 334             }
 335         }
 336         else if (*lc_data == '%')
 337             expand_prefix_found = TRUE;
 338         else
 339         {
 340             if (!whitespace (*lc_data))
 341                 written_nonspace = TRUE;
 342             if (is_cd)
 343                 *(pbuffer++) = *lc_data;
 344             else
 345                 g_string_append_c (shell_string, *lc_data);
 346         }
 347     }                           /* for */
 348     return g_string_free (shell_string, FALSE);
 349 }

 381 static void
 382 exec_extension_cd (void)
     /*  [+382 src/filemanager/ext.c] */
 383 {
 384     char *q;
 385     vfs_path_t *p_vpath;
 386 
 387     *pbuffer = '\0';
 388     pbuffer = buffer;
 389     /* Search last non-space character. Start search at the end in order
 390        not to short filenames containing spaces. */
 391     q = pbuffer + strlen (pbuffer) - 1;
 392     while (q >= pbuffer && whitespace (*q))
 393         q--;
 394     q[1] = 0;
 395 
 396     p_vpath = vfs_path_from_str_flags (pbuffer, VPF_NO_CANON);
 397     do_cd (p_vpath, cd_parse_command);
 398     vfs_path_free (p_vpath);
 399 }

 404 static vfs_path_t *
 405 exec_extension (void *target, const vfs_path_t * filename_vpath, const char *lc_data,
     /*  [+405 src/filemanager/ext.c] */
 406                 int start_line)
 407 {
 408     char *shell_string, *export_variables;
 409     vfs_path_t *script_vpath = NULL;
 410     int cmd_file_fd;
 411     FILE *cmd_file;
 412     char *cmd = NULL;
 413 
 414     g_return_val_if_fail (lc_data != NULL, NULL);
 415 
 416     pbuffer = NULL;
 417     localmtime = 0;
 418     quote_func = name_quote;
 419     run_view = FALSE;
 420     is_cd = FALSE;
 421     written_nonspace = FALSE;
 422 
 423     /* Avoid making a local copy if we are doing a cd */
 424     do_local_copy = !vfs_file_is_local (filename_vpath);
 425 
 426     shell_string = exec_make_shell_string (lc_data, filename_vpath);
 427 
 428     if (shell_string == NULL)
 429         goto ret;
 430 
 431     if (is_cd)
 432     {
 433         exec_extension_cd ();
 434         g_free (shell_string);
 435         goto ret;
 436     }
 437 
 438     /*
 439      * All commands should be run in /bin/sh regardless of user shell.
 440      * To do that, create temporary shell script and run it.
 441      * Sometimes it's not needed (e.g. for %cd and %view commands),
 442      * but it's easier to create it anyway.
 443      */
 444     cmd_file_fd = mc_mkstemps (&script_vpath, "mcext", SCRIPT_SUFFIX);
 445 
 446     if (cmd_file_fd == -1)
 447     {
 448         message (D_ERROR, MSG_ERROR,
 449                  _("Cannot create temporary command file\n%s"), unix_error_string (errno));
 450         goto ret;
 451     }
 452 
 453     cmd_file = fdopen (cmd_file_fd, "w");
 454     fputs ("#! /bin/sh\n\n", cmd_file);
 455 
 456     export_variables = exec_get_export_variables (filename_vpath);
 457     if (export_variables != NULL)
 458     {
 459         fprintf (cmd_file, "%s\n", export_variables);
 460         g_free (export_variables);
 461     }
 462 
 463     fputs (shell_string, cmd_file);
 464     g_free (shell_string);
 465 
 466     /*
 467      * Make the script remove itself when it finishes.
 468      * Don't do it for the viewer - it may need to rerun the script,
 469      * so we clean up after calling view().
 470      */
 471     if (!run_view)
 472         fprintf (cmd_file, "\n/bin/rm -f %s\n", vfs_path_as_str (script_vpath));
 473 
 474     fclose (cmd_file);
 475 
 476     if ((run_view && !written_nonspace) || is_cd)
 477     {
 478         exec_cleanup_script (script_vpath);
 479         script_vpath = NULL;
 480     }
 481     else
 482     {
 483         /* Set executable flag on the command file ... */
 484         mc_chmod (script_vpath, S_IRWXU);
 485         /* ... but don't rely on it - run /bin/sh explicitly */
 486         cmd = g_strconcat ("/bin/sh ", vfs_path_as_str (script_vpath), (char *) NULL);
 487     }
 488 
 489     if (run_view)
 490     {
 491         /* If we've written whitespace only, then just load filename into view */
 492         if (!written_nonspace)
 493             exec_extension_view (target, NULL, filename_vpath, start_line);
 494         else
 495             exec_extension_view (target, cmd, filename_vpath, start_line);
 496     }
 497     else
 498     {
 499         shell_execute (cmd, EXECUTE_INTERNAL);
 500         if (mc_global.tty.console_flag != '\0')
 501         {
 502             handle_console (CONSOLE_SAVE);
 503             if (output_lines && mc_global.keybar_visible)
 504                 show_console_contents (output_start_y,
 505                                        LINES - mc_global.keybar_visible -
 506                                        output_lines - 1, LINES - mc_global.keybar_visible - 1);
 507         }
 508     }
 509 
 510     g_free (cmd);
 511 
 512     exec_cleanup_file_name (filename_vpath, TRUE);
 513   ret:
 514     return script_vpath;
 515 }

 518 /**
 519  * Run cmd_file with args, put result into buf.
 520  * If error, put '\0' into buf[0]
 521  * Return 1 if the data is valid, 0 otherwise, -1 for fatal errors.
 522  *
 523  * NOTES: buf is null-terminated string.
 524  */

 526 static int
 527 get_popen_information (const char *cmd_file, const char *args, char *buf, int buflen)
     /*  [+527 src/filemanager/ext.c] */
 528 {
 529     gboolean read_bytes = FALSE;
 530     char *command;
 531     FILE *f;
 532 
 533     command = g_strconcat (cmd_file, args, " 2>/dev/null", (char *) NULL);
 534     f = popen (command, "r");
 535     g_free (command);
 536 
 537     if (f != NULL)
 538     {
 539 #ifdef __QNXNTO__
 540         if (setvbuf (f, NULL, _IOFBF, 0) != 0)
 541         {
 542             (void) pclose (f);
 543             return -1;
 544         }
 545 #endif
 546         read_bytes = (fgets (buf, buflen, f) != NULL);
 547         if (!read_bytes)
 548             buf[0] = '\0';      /* Paranoid termination */
 549         pclose (f);
 550     }
 551     else
 552     {
 553         buf[0] = '\0';          /* Paranoid termination */
 554         return -1;
 555     }
 556 
 557     buf[buflen - 1] = '\0';
 558 
 559     return read_bytes ? 1 : 0;
 560 }
