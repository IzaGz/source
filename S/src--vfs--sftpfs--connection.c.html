  67 sftpfs_open_socket (struct vfs_s_super *super, GError ** mcerror)
     /*  [+67 src/vfs/sftpfs/connection.c] */
  68 {
  69     struct addrinfo hints, *res = NULL, *curr_res;
  70     int my_socket = 0;
  71     char port[BUF_TINY];
  72     int e;
  73 
  74     mc_return_val_if_error (mcerror, LIBSSH2_INVALID_SOCKET);
  75 
  76     if (super->path_element->host == NULL || *super->path_element->host == '\0')
  77     {
  78         mc_propagate_error (mcerror, 0, "%s", _("sftp: Invalid host name."));
  79         return LIBSSH2_INVALID_SOCKET;
  80     }
  81 
  82     sprintf (port, "%hu", (unsigned short) super->path_element->port);
  83 
  84     tty_enable_interrupt_key ();        /* clear the interrupt flag */
  85 
  86     memset (&hints, 0, sizeof (hints));
  87     hints.ai_family = AF_UNSPEC;
  88     hints.ai_socktype = SOCK_STREAM;
  89 
  90 #ifdef AI_ADDRCONFIG
  91     /* By default, only look up addresses using address types for
  92      * which a local interface is configured (i.e. no IPv6 if no IPv6
  93      * interfaces, likewise for IPv4 (see RFC 3493 for details). */
  94     hints.ai_flags = AI_ADDRCONFIG;
  95 #endif
  96 
  97     e = getaddrinfo (super->path_element->host, port, &hints, &res);
  98 
  99 #ifdef AI_ADDRCONFIG
 100     if (e == EAI_BADFLAGS)
 101     {
 102         /* Retry with no flags if AI_ADDRCONFIG was rejected. */
 103         hints.ai_flags = 0;
 104         e = getaddrinfo (super->path_element->host, port, &hints, &res);
 105     }
 106 #endif
 107 
 108     if (e != 0)
 109     {
 110         mc_propagate_error (mcerror, e, _("sftp: %s"), gai_strerror (e));
 111         my_socket = LIBSSH2_INVALID_SOCKET;
 112         goto ret;
 113     }
 114 
 115     for (curr_res = res; curr_res != NULL; curr_res = curr_res->ai_next)
 116     {
 117         int save_errno;
 118 
 119         my_socket = socket (curr_res->ai_family, curr_res->ai_socktype, curr_res->ai_protocol);
 120 
 121         if (my_socket < 0)
 122         {
 123             if (curr_res->ai_next != NULL)
 124                 continue;
 125 
 126             vfs_print_message (_("sftp: %s"), unix_error_string (errno));
 127             my_socket = LIBSSH2_INVALID_SOCKET;
 128             goto ret;
 129         }
 130 
 131         vfs_print_message (_("sftp: making connection to %s"), super->path_element->host);
 132 
 133         if (connect (my_socket, curr_res->ai_addr, curr_res->ai_addrlen) >= 0)
 134             break;
 135 
 136         save_errno = errno;
 137 
 138         close (my_socket);
 139 
 140         if (save_errno == EINTR && tty_got_interrupt ())
 141             mc_propagate_error (mcerror, 0, "%s", _("sftp: connection interrupted by user"));
 142         else if (res->ai_next == NULL)
 143             mc_propagate_error (mcerror, save_errno, _("sftp: connection to server failed: %s"),
 144                                 unix_error_string (save_errno));
 145         else
 146             continue;
 147 
 148         my_socket = LIBSSH2_INVALID_SOCKET;
 149         break;
 150     }
 151 
 152   ret:
 153     if (res != NULL)
 154         freeaddrinfo (res);
 155     tty_disable_interrupt_key ();
 156     return my_socket;
 157 }

     /*  [+167 src/vfs/sftpfs/connection.c] */
 168 {
 169     char *userauthlist;
 170     sftpfs_super_data_t *super_data;
 171 
 172     super_data = (sftpfs_super_data_t *) super->data;
 173 
 174     super_data->auth_type = NONE;
 175 
 176     /* check what authentication methods are available */
 177     /* userauthlist is internally managed by libssh2 and freed by libssh2_session_free() */
 178     userauthlist = libssh2_userauth_list (super_data->session, super->path_element->user,
 179                                           strlen (super->path_element->user));
 180 
 181     if ((strstr (userauthlist, "password") != NULL
 182          || strstr (userauthlist, "keyboard-interactive") != NULL)
 183         && (super_data->config_auth_type & PASSWORD) != 0)
 184         super_data->auth_type |= PASSWORD;
 185 
 186     if (strstr (userauthlist, "publickey") != NULL && (super_data->config_auth_type & PUBKEY) != 0)
 187         super_data->auth_type |= PUBKEY;
 188 
 189     if ((super_data->config_auth_type & AGENT) != 0)
 190         super_data->auth_type |= AGENT;
 191 }

 202 static gboolean
 203 sftpfs_open_connection_ssh_agent (struct vfs_s_super *super, GError ** mcerror)
     /*  [+203 src/vfs/sftpfs/connection.c] */
 204 {
 205     sftpfs_super_data_t *super_data;
 206     struct libssh2_agent_publickey *identity, *prev_identity = NULL;
 207     int rc;
 208 
 209     mc_return_val_if_error (mcerror, FALSE);
 210 
 211     super_data = (sftpfs_super_data_t *) super->data;
 212     super_data->agent = NULL;
 213 
 214     if ((super_data->auth_type & AGENT) == 0)
 215         return FALSE;
 216 
 217     /* Connect to the ssh-agent */
 218     super_data->agent = libssh2_agent_init (super_data->session);
 219     if (super_data->agent == NULL)
 220         return FALSE;
 221 
 222     if (libssh2_agent_connect (super_data->agent) != 0)
 223         return FALSE;
 224 
 225     if (libssh2_agent_list_identities (super_data->agent) != 0)
 226         return FALSE;
 227 
 228     while (TRUE)
 229     {
 230         rc = libssh2_agent_get_identity (super_data->agent, &identity, prev_identity);
 231         if (rc == 1)
 232             break;
 233 
 234         if (rc < 0)
 235             return FALSE;
 236 
 237         if (libssh2_agent_userauth (super_data->agent, super->path_element->user, identity) == 0)
 238             break;
 239 
 240         prev_identity = identity;
 241     }
 242 
 243     return (rc == 0);
 244 }

 286                                                           super->path_element->user,
 287                                                           super_data->pubkey, super_data->privkey,
 288                                                           passwd) == 0);

     /*  [+305 src/vfs/sftpfs/connection.c] */
 306 {
 307     sftpfs_super_data_t *super_data;
 308     char *p, *passwd;
 309     gboolean ret_value = FALSE;
 310     int rc;
 311 
 312     mc_return_val_if_error (mcerror, FALSE);
 313 
 314     super_data = (sftpfs_super_data_t *) super->data;
 315 
 316     if ((super_data->auth_type & PASSWORD) == 0)
 317         return FALSE;
 318 
 319     if (super->path_element->password != NULL)
 320     {
 321         while ((rc = libssh2_userauth_password (super_data->session, super->path_element->user,
 322                                                 super->path_element->password)) ==

 324         if (rc == 0)
 325             return TRUE;
 326     }
 327 
 328     p = g_strdup_printf (_("sftp: Enter password for %s "), super->path_element->user);
 329     passwd = vfs_get_password (p);
 330     g_free (p);
 331 
 332     if (passwd == NULL)
 333         mc_propagate_error (mcerror, 0, "%s", _("sftp: Password is empty."));
 334     else
 335     {
 336         while ((rc = libssh2_userauth_password (super_data->session, super->path_element->user,
 337                                                 passwd)) == LIBSSH2_ERROR_EAGAIN)
 338             ;
 339 
 340         if (rc == 0)
 341         {
 342             ret_value = TRUE;
 343             g_free (super->path_element->password);
 344             super->path_element->password = passwd;
 345         }
 346         else
 347             g_free (passwd);
 348     }
 349 
 350     return ret_value;
 351 }

 364 int
 365 sftpfs_open_connection (struct vfs_s_super *super, GError ** mcerror)
     /*  [+365 src/vfs/sftpfs/connection.c] */
 366 {
 367     int rc;
 368     sftpfs_super_data_t *super_data;
 369 
 370     mc_return_val_if_error (mcerror, -1);
 371 
 372     super_data = (sftpfs_super_data_t *) super->data;
 373 
 374     /*
 375      * The application code is responsible for creating the socket
 376      * and establishing the connection
 377      */
 378     super_data->socket_handle = sftpfs_open_socket (super, mcerror);
 379     if (super_data->socket_handle == LIBSSH2_INVALID_SOCKET)
 380         return (-1);
 381 
 382     /* Create a session instance */
 383     super_data->session = libssh2_session_init ();
 384     if (super_data->session == NULL)
 385         return (-1);
 386 
 387     /* ... start it up. This will trade welcome banners, exchange keys,
 388      * and setup crypto, compression, and MAC layers
 389      */
 390 #if LIBSSH2_VERSION_NUM < 0x010208
 391     rc = libssh2_session_startup (super_data->session, super_data->socket_handle);
 392 #else
 393     rc = libssh2_session_handshake (super_data->session,
 394                                     (libssh2_socket_t) super_data->socket_handle);
 395 #endif
 396     if (rc != 0)
 397     {
 398         mc_propagate_error (mcerror, rc, "%s", _("sftp: Failure establishing SSH session"));
 399         return (-1);
 400     }
 401 
 402     /* At this point we havn't yet authenticated.  The first thing to do
 403      * is check the hostkey's fingerprint against our known hosts Your app
 404      * may have it hard coded, may go to a file, may present it to the
 405      * user, that's your call
 406      */
 407     super_data->fingerprint = libssh2_hostkey_hash (super_data->session, LIBSSH2_HOSTKEY_HASH_SHA1);
 408 
 409     sftpfs_recognize_auth_types (super);
 410 
 411     if (!sftpfs_open_connection_ssh_agent (super, mcerror)
 412         && !sftpfs_open_connection_ssh_key (super, mcerror)
 413         && !sftpfs_open_connection_ssh_password (super, mcerror))
 414         return (-1);
 415 
 416     super_data->sftp_session = libssh2_sftp_init (super_data->session);
 417 
 418     if (super_data->sftp_session == NULL)
 419         return (-1);
 420 
 421     /* Since we have not set non-blocking, tell libssh2 we are blocking */
 422     libssh2_session_set_blocking (super_data->session, 1);
 423 
 424     return 0;
 425 }

 436 void
 437 sftpfs_close_connection (struct vfs_s_super *super, const char *shutdown_message, GError ** mcerror)
     /*  [+437 src/vfs/sftpfs/connection.c] */
 438 {
 439     sftpfs_super_data_t *super_data;
 440 
 441     /* no mc_return_*_if_error() here because of abort open_connection handling too */
 442     (void) mcerror;
 443 
 444     super_data = (sftpfs_super_data_t *) super->data;
 445     if (super_data == NULL)
 446         return;
 447 
 448     if (super_data->sftp_session != NULL)
 449     {
 450         libssh2_sftp_shutdown (super_data->sftp_session);
 451         super_data->sftp_session = NULL;
 452     }
 453 
 454     if (super_data->agent != NULL)
 455     {
 456         libssh2_agent_disconnect (super_data->agent);
 457         libssh2_agent_free (super_data->agent);
 458         super_data->agent = NULL;
 459     }
 460 
 461     super_data->fingerprint = NULL;
 462 
 463     if (super_data->session != NULL)
 464     {
 465         libssh2_session_disconnect (super_data->session, shutdown_message);
 466         libssh2_session_free (super_data->session);
 467         super_data->session = NULL;
 468     }
 469 
 470     if (super_data->socket_handle != LIBSSH2_INVALID_SOCKET)
 471     {
 472         close (super_data->socket_handle);
 473         super_data->socket_handle = LIBSSH2_INVALID_SOCKET;
 474     }
 475 }
