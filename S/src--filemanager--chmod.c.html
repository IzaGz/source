 135 static void
 136 chmod_i18n (void)
     /*  [+136 src/filemanager/chmod.c] */
 137 {
 138     static gboolean i18n = FALSE;
 139     int i, len;
 140 
 141     if (i18n)
 142         return;
 143 
 144     i18n = TRUE;
 145 
 146 #ifdef ENABLE_NLS
 147     for (i = 0; i < check_perm_num; i++)
 148         check_perm[i].text = _(check_perm[i].text);
 149 
 150     for (i = 0; i < file_info_labels_num; i++)
 151         file_info_labels[i] = _(file_info_labels[i]);
 152 
 153     for (i = 0; i < chmod_but_num; i++)
 154         chmod_but[i].text = _(chmod_but[i].text);
 155 #endif /* ENABLE_NLS */
 156 
 157     for (i = 0; i < check_perm_num; i++)
 158     {
 159         len = str_term_width1 (check_perm[i].text);
 160         check_perm_len = MAX (check_perm_len, len);
 161     }
 162 
 163     check_perm_len += 1 + 3 + 1;        /* mark, [x] and space */
 164 
 165     for (i = 0; i < file_info_labels_num; i++)
 166     {
 167         len = str_term_width1 (file_info_labels[i]) + 2;        /* spaces around */
 168         file_info_labels_len = MAX (file_info_labels_len, len);
 169     }
 170 
 171     for (i = 0; i < chmod_but_num; i++)
 172     {
 173         chmod_but[i].len = str_term_width1 (chmod_but[i].text) + 3;     /* [], spaces and w/o & */
 174         if (chmod_but[i].flags == DEFPUSH_BUTTON)
 175             chmod_but[i].len += 2;      /* <> */
 176     }
 177 }

 216 static cb_ret_t
 217 chmod_callback (Widget * w, Widget * sender, widget_msg_t msg, int parm, void *data)
     /*  [+217 src/filemanager/chmod.c] */
 218 {
 219     WDialog *h = DIALOG (w);
 220 
 221     switch (msg)
 222     {
 223     case MSG_NOTIFY:
 224         {
 225             /* handle checkboxes */
 226             int i;
 227 
 228             /* whether notification was sent by checkbox? */
 229             for (i = 0; i < check_perm_num; i++)
 230                 if (sender == WIDGET (check_perm[i].check))
 231                     break;
 232 
 233             if (i < check_perm_num)
 234             {
 235                 char buffer[BUF_TINY];
 236 
 237                 c_stat ^= check_perm[i].mode;
 238                 g_snprintf (buffer, sizeof (buffer), "%o", (unsigned int) c_stat);
 239                 label_set_text (statl, buffer);
 240                 chmod_toggle_select (h, i);
 241                 mode_change = TRUE;
 242                 return MSG_HANDLED;
 243             }
 244         }
 245 
 246         return MSG_NOT_HANDLED;
 247 
 248     case MSG_KEY:
 249         if (parm == 'T' || parm == 't' || parm == KEY_IC)
 250         {
 251             int i;
 252             unsigned long id;
 253 
 254             id = dlg_get_current_widget_id (h);
 255             for (i = 0; i < check_perm_num; i++)
 256                 if (id == WIDGET (check_perm[i].check)->id)
 257                     break;
 258 
 259             if (i < check_perm_num)
 260             {
 261                 chmod_toggle_select (h, i);
 262                 if (parm == KEY_IC)
 263                     dlg_select_next_widget (h);
 264                 return MSG_HANDLED;
 265             }
 266         }
 267         return MSG_NOT_HANDLED;
 268 
 269     case MSG_DRAW:
 270         chmod_refresh (h);
 271         return MSG_HANDLED;
 272 
 273     default:
 274         return dlg_default_callback (w, sender, msg, parm, data);
 275     }
 276 }

 395 static void
 396 do_chmod (struct stat *sf)
     /*  [+396 src/filemanager/chmod.c] */
 397 {
 398     vfs_path_t *vpath;
 399     sf->st_mode &= and_mask;
 400     sf->st_mode |= or_mask;
 401 
 402     vpath = vfs_path_from_str (current_panel->dir.list[c_file].fname);
 403     if (mc_chmod (vpath, sf->st_mode) == -1)
 404         message (D_ERROR, MSG_ERROR, _("Cannot chmod \"%s\"\n%s"),
 405                  current_panel->dir.list[c_file].fname, unix_error_string (errno));
 406 
 407     vfs_path_free (vpath);
 408     do_file_mark (current_panel, c_file, 0);
 409 }

 413 static void
 414 apply_mask (struct stat *sf)
     /*  [+414 src/filemanager/chmod.c] */
 415 {
 416     need_update = TRUE;
 417     end_chmod = TRUE;
 418 
 419     do_chmod (sf);
 420 
 421     do
 422     {
 423         char *fname;
 424         vfs_path_t *vpath;
 425         gboolean ok;
 426 
 427         fname = next_file ();
 428         vpath = vfs_path_from_str (fname);
 429         ok = (mc_stat (vpath, sf) == 0);
 430         vfs_path_free (vpath);
 431         if (!ok)
 432             return;
 433 
 434         c_stat = sf->st_mode;
 435 
 436         do_chmod (sf);
 437     }
 438     while (current_panel->marked != 0);
 439 }

 445 void
 446 chmod_cmd (void)
     /*  [+446 src/filemanager/chmod.c] */
 447 {
 448     chmod_i18n ();
 449 
 450     do
 451     {                           /* do while any files remaining */
 452         vfs_path_t *vpath;
 453         WDialog *ch_dlg;
 454         struct stat sf_stat;
 455         char *fname;
 456         int i, result;
 457 
 458         do_refresh ();
 459 
 460         mode_change = FALSE;
 461         need_update = FALSE;
 462         end_chmod = FALSE;
 463         c_file = 0;
 464 
 465         if (current_panel->marked != 0)
 466             fname = next_file ();       /* next marked file */
 467         else
 468             fname = selection (current_panel)->fname;   /* single file */
 469 
 470         vpath = vfs_path_from_str (fname);
 471 
 472         if (mc_stat (vpath, &sf_stat) != 0)
 473         {
 474             vfs_path_free (vpath);
 475             break;
 476         }
 477 
 478         c_stat = sf_stat.st_mode;
 479 
 480         ch_dlg = init_chmod (fname, &sf_stat);
 481 
 482         /* do action */
 483         result = dlg_run (ch_dlg);
 484 
 485         switch (result)
 486         {
 487         case B_ENTER:
 488             if (mode_change && mc_chmod (vpath, c_stat) == -1)
 489                 message (D_ERROR, MSG_ERROR, _("Cannot chmod \"%s\"\n%s"),
 490                          fname, unix_error_string (errno));
 491             need_update = TRUE;
 492             break;
 493 
 494         case B_CANCEL:
 495             end_chmod = TRUE;
 496             break;
 497 
 498         case B_ALL:
 499         case B_MARKED:
 500             and_mask = or_mask = 0;
 501             and_mask = ~and_mask;
 502 
 503             for (i = 0; i < check_perm_num; i++)
 504                 if (check_perm[i].selected || result == B_ALL)
 505                 {
 506                     if (check_perm[i].check->state)
 507                         or_mask |= check_perm[i].mode;
 508                     else
 509                         and_mask &= ~check_perm[i].mode;
 510                 }
 511 
 512             apply_mask (&sf_stat);
 513             break;
 514 
 515         case B_SETMRK:
 516             and_mask = or_mask = 0;
 517             and_mask = ~and_mask;
 518 
 519             for (i = 0; i < check_perm_num; i++)
 520                 if (check_perm[i].selected)
 521                     or_mask |= check_perm[i].mode;
 522 
 523             apply_mask (&sf_stat);
 524             break;
 525 
 526         case B_CLRMRK:
 527             and_mask = or_mask = 0;
 528             and_mask = ~and_mask;
 529 
 530             for (i = 0; i < check_perm_num; i++)
 531                 if (check_perm[i].selected)
 532                     and_mask &= ~check_perm[i].mode;
 533 
 534             apply_mask (&sf_stat);
 535             break;
 536 
 537         default:
 538             break;
 539         }
 540 
 541         if (current_panel->marked != 0 && result != B_CANCEL)
 542         {
 543             do_file_mark (current_panel, c_file, 0);
 544             need_update = TRUE;
 545         }
 546 
 547         vfs_path_free (vpath);
 548 
 549         dlg_destroy (ch_dlg);
 550     }
 551     while (current_panel->marked != 0 && !end_chmod);
 552 
 553     chmod_done ();
 554 }
