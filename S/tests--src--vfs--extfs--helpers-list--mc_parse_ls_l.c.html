 176 static void
 177 chomp (char *s)
     /*  [+177 tests/src/vfs/extfs/helpers-list/mc_parse_ls_l.c] */
 178 {
 179     int i;
 180 
 181     i = strlen (s);
 182 
 183     /* Code taken from vfs_parse_ls_lga(), with modifications. */
 184     if ((--i >= 0) && (s[i] == '\r' || s[i] == '\n'))
 185         s[i] = '\0';
 186     if ((--i >= 0) && (s[i] == '\r' || s[i] == '\n'))
 187         s[i] = '\0';
 188 }

     /*  [+193 tests/src/vfs/extfs/helpers-list/mc_parse_ls_l.c] */

 195     static char buf[BUF_SMALL];

 237 static void
 238 yaml_dump_stbuf (const struct stat *st)
     /*  [+238 tests/src/vfs/extfs/helpers-list/mc_parse_ls_l.c] */
 239 {
 240     /* Various casts and flags here were taken/inspired by info_show_info(). */
 241     printf ("  perm: %s\n", string_perm (st->st_mode));
 242     if (!opt_drop_ids)
 243     {
 244         printf ("  uid: %s\n", symbolic_uid (st->st_uid));
 245         printf ("  gid: %s\n", symbolic_gid (st->st_gid));
 246     }
 247     printf ("  size: %" PRIuMAX "\n", (uintmax_t) st->st_size);
 248     printf ("  nlink: %d\n", (int) st->st_nlink);
 249     if (!opt_drop_mtime)
 250         printf ("  mtime: %s\n", string_date (st->st_mtime));
 251 }

 295 {
 296     /* Various casts and flags here were taken/inspired by info_show_info(). */
 297     printf ("%s %3d ", string_perm (st->st_mode), (int) st->st_nlink);

 335 static void
 336 process_ls_line (const char *line)
     /*  [+336 tests/src/vfs/extfs/helpers-list/mc_parse_ls_l.c] */
 337 {
 338     struct stat st;
 339     char *filename, *linkname;
 340     gboolean success;
 341 
 342     memset (&st, 0, sizeof st);
 343     filename = NULL;
 344     linkname = NULL;
 345 
 346     success = vfs_parse_ls_lga (line, &st, &filename, &linkname, NULL);
 347 
 348     if (!success)
 349         error_count++;
 350 
 351     if (opt_output_format == FORMAT_YAML)
 352         yaml_dump_record (success, line, &st, filename, linkname);
 353     else
 354         ls_dump_record (success, line, &st, filename, linkname);
 355 
 356     g_free (filename);
 357     g_free (linkname);
 358 }

 362 static void
 363 process_input (FILE * input)
     /*  [+363 tests/src/vfs/extfs/helpers-list/mc_parse_ls_l.c] */
 364 {
 365     char line[BUF_4K];
 366 
 367     while (fgets (line, sizeof line, input) != NULL)
 368     {
 369         chomp (line);           /* Not mandatory. Makes error messages nicer. */
 370         if (strncmp (line, "total ", 6) == 0)   /* Convenience only: makes 'ls -l' parse cleanly. */
 371             continue;
 372         process_ls_line (line);
 373     }
 374 }

 378 int
 379 main (int argc, char *argv[])
     /*  [+379 tests/src/vfs/extfs/helpers-list/mc_parse_ls_l.c] */
 380 {
 381     FILE *input;
 382 
 383     if (!parse_command_line (&argc, &argv))
 384         return EXIT_FAILURE;
 385 
 386     if (argc >= 2)
 387     {
 388         input = fopen (argv[1], "r");
 389         if (input == NULL)
 390         {
 391             perror (argv[1]);
 392             return EXIT_FAILURE;
 393         }
 394     }
 395     else
 396     {
 397         input = stdin;
 398     }
 399 
 400     process_input (input);
 401 
 402     return (error_count > 0) ? EXIT_FAILURE : EXIT_SUCCESS;
 403 }
