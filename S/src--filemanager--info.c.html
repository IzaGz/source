  78 static void
  79 info_box (WInfo * info)
     /*  [+79 src/filemanager/info.c] */
  80 {
  81     Widget *w = WIDGET (info);
  82 
  83     const char *title = _("Information");
  84     const int len = str_term_width1 (title);
  85 
  86     tty_set_normal_attrs ();
  87     tty_setcolor (NORMAL_COLOR);
  88     widget_erase (w);
  89     tty_draw_box (w->y, w->x, w->lines, w->cols, FALSE);
  90 
  91     widget_move (w, 0, (w->cols - len - 2) / 2);
  92     tty_printf (" %s ", title);
  93 
  94     widget_move (w, 2, 0);
  95     tty_print_alt_char (ACS_LTEE, FALSE);
  96     widget_move (w, 2, w->cols - 1);
  97     tty_print_alt_char (ACS_RTEE, FALSE);
  98     tty_draw_hline (w->y + 2, w->x + 1, ACS_HLINE, w->cols - 2);
  99 }

 103 static void
 104 info_show_info (WInfo * info)
     /*  [+104 src/filemanager/info.c] */
 105 {
 106     Widget *w = WIDGET (info);
 107     static int i18n_adjust = 0;
 108     static const char *file_label;
 109     GString *buff;
 110     struct stat st;
 111     char rp_cwd[PATH_MAX];
 112     const char *p_rp_cwd;
 113 
 114     if (!is_idle ())
 115         return;
 116 
 117     info_box (info);
 118 
 119     tty_setcolor (MARKED_COLOR);
 120     widget_move (w, 1, 3);
 121     tty_printf (_("Midnight Commander %s"), VERSION);
 122 
 123     if (!info->ready)
 124         return;
 125 
 126     if (get_current_type () != view_listing)
 127         return;
 128 
 129     /* don't rely on vpath CWD when cd_symlinks enabled */
 130     p_rp_cwd = mc_realpath (vfs_path_as_str (current_panel->cwd_vpath), rp_cwd);
 131     if (p_rp_cwd == NULL)
 132         p_rp_cwd = vfs_path_as_str (current_panel->cwd_vpath);
 133 
 134     my_statfs (&myfs_stats, p_rp_cwd);
 135 
 136     st = current_panel->dir.list[current_panel->selected].st;
 137 
 138     /* Print only lines which fit */
 139 
 140     if (i18n_adjust == 0)
 141     {
 142         /* This printf pattern string is used as a reference for size */
 143         file_label = _("File: %s");
 144         i18n_adjust = str_term_width1 (file_label) + 2;
 145     }
 146 
 147     tty_setcolor (NORMAL_COLOR);
 148 
 149     buff = g_string_new ("");
 150 
 151     switch (w->lines - 2)
 152     {
 153         /* Note: all cases are fall-throughs */
 154 
 155     default:
 156 
 157     case 16:
 158         widget_move (w, 16, 3);
 159         if ((myfs_stats.nfree == 0 && myfs_stats.nodes == 0) ||
 160             (myfs_stats.nfree == (uintmax_t) (-1) && myfs_stats.nodes == (uintmax_t) (-1)))
 161             tty_print_string (_("No node information"));
 162         else if (myfs_stats.nfree == (uintmax_t) (-1))
 163             tty_printf ("%s -/%" PRIuMAX, _("Free nodes:"), myfs_stats.nodes);
 164         else if (myfs_stats.nodes == (uintmax_t) (-1))
 165             tty_printf ("%s %" PRIuMAX "/-", _("Free nodes:"), myfs_stats.nfree);
 166         else
 167             tty_printf ("%s %" PRIuMAX "/%" PRIuMAX " (%d%%)",
 168                         _("Free nodes:"),
 169                         myfs_stats.nfree, myfs_stats.nodes,
 170                         myfs_stats.nodes == 0 ? 0 :
 171                         (int) (100 * (long double) myfs_stats.nfree / myfs_stats.nodes));
 172 
 173     case 15:
 174         widget_move (w, 15, 3);
 175         if (myfs_stats.avail == 0 && myfs_stats.total == 0)
 176             tty_print_string (_("No space information"));
 177         else
 178         {
 179             char buffer1[6], buffer2[6];
 180 
 181             size_trunc_len (buffer1, 5, myfs_stats.avail, 1, panels_options.kilobyte_si);
 182             size_trunc_len (buffer2, 5, myfs_stats.total, 1, panels_options.kilobyte_si);
 183             tty_printf (_("Free space: %s/%s (%d%%)"), buffer1, buffer2,
 184                         myfs_stats.total == 0 ? 0 :
 185                         (int) (100 * (long double) myfs_stats.avail / myfs_stats.total));
 186         }
 187 
 188     case 14:
 189         widget_move (w, 14, 3);
 190         tty_printf (_("Type:       %s"),
 191                     myfs_stats.typename ? myfs_stats.typename : _("non-local vfs"));
 192         if (myfs_stats.type != 0xffff && myfs_stats.type != -1)
 193             tty_printf (" (%Xh)", (unsigned int) myfs_stats.type);
 194 
 195     case 13:
 196         widget_move (w, 13, 3);
 197         str_printf (buff, _("Device:     %s"),
 198                     str_trunc (myfs_stats.device, w->cols - i18n_adjust));
 199         tty_print_string (buff->str);
 200         g_string_set_size (buff, 0);
 201     case 12:
 202         widget_move (w, 12, 3);
 203         str_printf (buff, _("Filesystem: %s"),
 204                     str_trunc (myfs_stats.mpoint, w->cols - i18n_adjust));
 205         tty_print_string (buff->str);
 206         g_string_set_size (buff, 0);
 207     case 11:
 208         widget_move (w, 11, 3);
 209         str_printf (buff, _("Accessed:   %s"), file_date (st.st_atime));
 210         tty_print_string (buff->str);
 211         g_string_set_size (buff, 0);
 212     case 10:
 213         widget_move (w, 10, 3);
 214         str_printf (buff, _("Modified:   %s"), file_date (st.st_mtime));
 215         tty_print_string (buff->str);
 216         g_string_set_size (buff, 0);
 217     case 9:
 218         widget_move (w, 9, 3);
 219         /* The field st_ctime is changed by writing or by setting inode
 220            information (i.e., owner, group, link count, mode, etc.).  */
 221         /* TRANSLATORS: Time of last status change as in stat(2) man. */
 222         str_printf (buff, _("Changed:    %s"), file_date (st.st_ctime));
 223         tty_print_string (buff->str);
 224         g_string_set_size (buff, 0);
 225 
 226     case 8:
 227         widget_move (w, 8, 3);
 228 #ifdef HAVE_STRUCT_STAT_ST_RDEV
 229         if (S_ISCHR (st.st_mode) || S_ISBLK (st.st_mode))
 230             tty_printf (_("Dev. type: major %lu, minor %lu"),
 231                         (unsigned long) major (st.st_rdev), (unsigned long) minor (st.st_rdev));
 232         else
 233 #endif
 234         {
 235             char buffer[10];
 236             size_trunc_len (buffer, 9, st.st_size, 0, panels_options.kilobyte_si);
 237             tty_printf (_("Size:       %s"), buffer);
 238 #ifdef HAVE_STRUCT_STAT_ST_BLOCKS
 239             tty_printf (ngettext (" (%lu block)", " (%lu blocks)",
 240                                   (unsigned long) st.st_blocks), (unsigned long) st.st_blocks);
 241 #endif
 242         }
 243 
 244     case 7:
 245         widget_move (w, 7, 3);
 246         tty_printf (_("Owner:      %s/%s"), get_owner (st.st_uid), get_group (st.st_gid));
 247 
 248     case 6:
 249         widget_move (w, 6, 3);
 250         tty_printf (_("Links:      %d"), (int) st.st_nlink);
 251 
 252     case 5:
 253         widget_move (w, 5, 3);
 254         tty_printf (_("Mode:       %s (%04o)"),
 255                     string_perm (st.st_mode), (unsigned) st.st_mode & 07777);
 256 
 257     case 4:
 258         widget_move (w, 4, 3);
 259         tty_printf (_("Location:   %Xh:%Xh"), (unsigned int) st.st_dev, (unsigned int) st.st_ino);
 260 
 261     case 3:
 262         {
 263             const char *fname;
 264 
 265             widget_move (w, 3, 2);
 266             fname = current_panel->dir.list[current_panel->selected].fname;
 267             str_printf (buff, file_label, str_trunc (fname, w->cols - i18n_adjust));
 268             tty_print_string (buff->str);
 269         }
 270 
 271     case 2:
 272     case 1:
 273     case 0:
 274         ;
 275     }                           /* switch */
 276     g_string_free (buff, TRUE);
 277 }

 281 static void
 282 info_hook (void *data)
     /*  [+282 src/filemanager/info.c] */
 283 {
 284     WInfo *info = (WInfo *) data;
 285     Widget *other_widget;
 286 
 287     other_widget = get_panel_widget (get_current_index ());
 288     if (!other_widget)
 289         return;
 290     if (widget_overlapped (WIDGET (info), other_widget))
 291         return;
 292 
 293     info->ready = TRUE;
 294     info_show_info (info);
 295 }

 299 static cb_ret_t
 300 info_callback (Widget * w, Widget * sender, widget_msg_t msg, int parm, void *data)
     /*  [+300 src/filemanager/info.c] */
 301 {
 302     WInfo *info = (WInfo *) w;
 303 
 304     switch (msg)
 305     {
 306     case MSG_INIT:
 307         init_my_statfs ();
 308         add_hook (&select_file_hook, info_hook, info);
 309         info->ready = FALSE;
 310         return MSG_HANDLED;
 311 
 312     case MSG_DRAW:
 313         info_hook (info);
 314         return MSG_HANDLED;
 315 
 316     case MSG_DESTROY:
 317         delete_hook (&select_file_hook, info_hook);
 318         free_my_statfs ();
 319         return MSG_HANDLED;
 320 
 321     default:
 322         return widget_default_callback (w, sender, msg, parm, data);
 323     }
 324 }
