  63 static void *
  64 local_open (const vfs_path_t * vpath, int flags, mode_t mode)
     /*  [+64 src/vfs/local/local.c] */
  65 {
  66     int *local_info;
  67     int fd;
  68     const vfs_path_element_t *path_element;
  69 
  70     path_element = vfs_path_get_by_index (vpath, -1);
  71     fd = open (path_element->path, NO_LINEAR (flags), mode);
  72     if (fd == -1)
  73         return 0;
  74 
  75     local_info = g_new (int, 1);
  76     *local_info = fd;
  77 
  78     return local_info;
  79 }

 223 static ssize_t
 224 local_write (void *data, const char *buf, size_t nbyte)
     /*  [+224 src/vfs/local/local.c] */
 225 {
 226     int fd;
 227     int n;
 228 
 229     if (!data)
 230         return -1;
 231 
 232     fd = *(int *) data;
 233     while ((n = write (fd, buf, nbyte)) == -1)
 234     {
 235 #ifdef EAGAIN
 236         if (errno == EAGAIN)
 237             continue;
 238 #endif
 239 #ifdef EINTR
 240         if (errno == EINTR)
 241             continue;
 242 #endif
 243         break;
 244     }
 245     return n;
 246 }

     /*  [+354 src/vfs/local/local.c] */
 355 {
 356     int n;
 357 
 358     if (!data)
 359         return -1;
 360 
 361     while ((n = read (*((int *) data), buffer, count)) == -1)
 362     {
 363 #ifdef EAGAIN
 364         if (errno == EAGAIN)
 365             continue;
 366 #endif
 367 #ifdef EINTR
 368         if (errno == EINTR)
 369             continue;
 370 #endif
 371         return -1;
 372     }
 373     return n;
