 139 static inline gboolean
 140 mcview_is_nroff_sequence (WView * view, off_t offset)
     /*  [+140 src/viewer/inlines.h] */
 141 {
 142     int c0, c1, c2;
 143 
 144     /* The following commands are ordered to speed up the calculation. */
 145 
 146     if (!mcview_get_byte_indexed (view, offset, 1, &c1) || c1 != '\b')
 147         return FALSE;
 148 
 149     if (!mcview_get_byte_indexed (view, offset, 0, &c0) || !g_ascii_isprint (c0))
 150         return FALSE;
 151 
 152     if (!mcview_get_byte_indexed (view, offset, 2, &c2) || !g_ascii_isprint (c2))
 153         return FALSE;
 154 
 155     return (c0 == c2 || c0 == '_' || (c0 == '+' && c2 == 'o'));
 156 }
