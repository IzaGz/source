 278 /* Many space usage primitives use all 1 bits to denote a value that is
 279    not applicable or unknown.  Propagate this information by returning
 280    a uintmax_t value that is all 1 bits if X is all 1 bits, even if X
 281    is unsigned and narrower than uintmax_t.  */
 282 #define PROPAGATE_ALL_ONES(x) \
 283   ((sizeof (x) < sizeof (uintmax_t) \
 284     && (~ (x) == (sizeof (x) < sizeof (int) \
 285           ? - (1 << (sizeof (x) * CHAR_BIT)) \
 286           : 0))) \
 287    ? UINTMAX_MAX : (uintmax_t) (x))
 288 
 289 /* Extract the top bit of X as an uintmax_t value.  */
 290 #define EXTRACT_TOP_BIT(x) ((x) & ((uintmax_t) 1 << (sizeof (x) * CHAR_BIT - 1)))
 291 
 292 /* If a value is negative, many space usage primitives store it into an
 293    integer variable by assignment, even if the variable's type is unsigned.
 294    So, if a space usage variable X's top bit is set, convert X to the
 295    uintmax_t value V such that (- (uintmax_t) V) is the negative of
 296    the original value.  If X's top bit is clear, just yield X.
 297    Use PROPAGATE_TOP_BIT if the original value might be negative;
 298    otherwise, use PROPAGATE_ALL_ONES.  */

 621 unescape_tab (char *str)
     /*  [+621 src/filemanager/mountlist.c] */
 622 {
 623     size_t i, j = 0;
 624     size_t len;
 625 
 626     len = strlen (str) + 1;
 627 
 628     for (i = 0; i < len; i++)
 629     {
 630         if (str[i] == '\\' && (i + 4 < len)
 631             && str[i + 1] >= '0' && str[i + 1] <= '3'
 632             && str[i + 2] >= '0' && str[i + 2] <= '7' && str[i + 3] >= '0' && str[i + 3] <= '7')
 633         {
 634             str[j++] = (str[i + 1] - '0') * 64 + (str[i + 2] - '0') * 8 + (str[i + 3] - '0');

 636         }
 637         else
 638             str[j++] = str[i];
